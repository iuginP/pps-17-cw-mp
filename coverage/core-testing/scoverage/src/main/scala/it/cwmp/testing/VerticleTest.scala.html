24     * Deploys all the verticles passed as parameter. If any deploy fails or isn't deployed withing the
25     * requested time, it throws a [[RuntimeException]]
26     *
27     * @param verticles the list of verticle to deploy
28     * @param atMost    the maximum amount of time we can wait for the verticle to start. By default: 10000 milliseconds
29     * @throws RuntimeException if any verticle can't be deployed
30     */
31   protected def deployAll(verticles: Traversable[ScalaVerticle], atMost: Duration = 10000.millis): Unit = {
32     deploymentIds = Set()
33     verticles.foreach(verticle => deploymentIds = deploymentIds + Await.result(vertx.deployVerticleFuture(verticle)
34       .andThen {
35         case Success(d) => d
36         case Failure(t) => throw new RuntimeException(t)
37       }, atMost))
38   }
39 
40   /**

42     * requested time, it throws a [[RuntimeException]]
43     *
44     * @param atMost the maximum amount of time we can wait for the verticle to stop. By default: 10000 milliseconds
45     * @throws RuntimeException if any verticle can't be un-deployed
46     */
47   protected def undeployAll(atMost: Duration = 10000.millis): Unit =
48     deploymentIds.foreach(id => Await.result(vertx.undeployFuture(id)
49       .andThen {
50         case Success(d) => d
51         case Failure(t) => throw new RuntimeException(t)
52       }, atMost))
