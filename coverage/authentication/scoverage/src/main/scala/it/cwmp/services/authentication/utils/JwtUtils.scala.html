19   /**
20     * Buid a token from a claim
21     * @param claim is the text that will use to build the token
22     * @return the token to use in the request
23     */
24   def encodeToken(claim: JwtClaim): Option[String] =
25     if (claim != null) Some(Jwt.encode(claim, secretKey, algorithm))
26     else None
27 
28   /**
29     * Decode a token
30     * @param token the token to decode
31     * @return the content of token decoded
32     */
33   def decodeToken(token: String): Option[JwtClaim] = {
34     Jwt.decodeRawAll(token, secretKey, Seq(algorithm)) match {
35       case Success((_, b, _)) =>
36         Some(JwtClaim(b))
37       case Failure(_) => None
38     }
39   }
40 
41   /**
42     * Validate a token
43     * @param token token to validate
44     * @return true if token is valid, false in opposite case
45     */
46   def validateToken(token: String): Boolean = {
47     Jwt.isValid(token, secretKey, Seq(algorithm))
48   }
49 
50   /**
51     * Build the token from the username
52     * @param username username to encode
53     * @return the token to use in the request
54     */
55   def encodeUsernameToken(username: String): Option[String] = {
56     if (username == null) None
57     else encodeToken(JwtClaim(new JsonObject().put(USERNAME_FIELD_NAME, username).encode()))
58   }
59 
60   /**
61     * Get the username from the token
62     * @param token token to decode
63     * @return user's username
64     */
65   def decodeUsernameToken(token: String): Option[String] = {
66     decodeToken(token).map(decoded => new JsonObject(decoded.content).getString(USERNAME_FIELD_NAME))
67   }
