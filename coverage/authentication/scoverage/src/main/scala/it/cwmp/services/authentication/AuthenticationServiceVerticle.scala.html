34   private def handlerSignUp: Handler[RoutingContext] = implicit routingContext => {
35     log.debug("Received sign up request.")
36     (for (
37       authorizationHeader <- request.getAuthenticationHeader;
38       (username, password) <- HttpUtils.readBasicAuthentication(authorizationHeader)
39     ) yield {
40       storageFuture flatMap (_.signUpFuture(username, password)) onComplete {
41         case Success(_) =>
42           log.info(s"User $username signed up.")
43           JwtUtils
44             .encodeUsernameToken(username)
45             .foreach(sendResponse(201, _))
46         case Failure(_) => sendResponse(400)
47       }
48     }) orElse Some(sendResponse(400))
49   }
50 
51   private def handlerSignOut: Handler[RoutingContext] = implicit routingContext => {
52     log.debug("Received sign out request.")
53     (for (
54       authorizationHeader <- request.getAuthenticationHeader;
55       token <- HttpUtils.readJwtAuthentication(authorizationHeader);
56       username <- JwtUtils.decodeUsernameToken(token)
57     ) yield {
58       // If every check pass, username contains the username contained in the token and we can check it exists
59       storageFuture.map(_.signOutFuture(username).onComplete {
60         case Success(_) =>
61           log.info(s"User $username signed out.")
62           sendResponse(202)
63         case Failure(_) => sendResponse(401)
64       })
65     }) orElse Some(sendResponse(400))
66   }
67 
68   private def handlerLogin: Handler[RoutingContext] = implicit routingContext => {
69     log.debug("Received login request.")

85   private def handlerValidation: Handler[RoutingContext] = implicit routingContext => {
86     log.debug("Received token validation request.")
87     (for (
88       authorizationHeader <- request.getAuthenticationHeader;
89       token <- HttpUtils.readJwtAuthentication(authorizationHeader);
90       username <- JwtUtils.decodeUsernameToken(token)
91     ) yield {
92       // If every check pass, username contains the username contained in the token and we can check it exists
93       storageFuture.map(_.existsFuture(username).onComplete {
94         case Success(_) =>
95           log.info(s"Token validation for $username successful")
96           sendResponse(200, username)
97         case Failure(_) => sendResponse(401)
98       })
99     }) orElse Some(sendResponse(400))
100   }
