35   private def handlerSignUp: Handler[RoutingContext] = implicit routingContext => {
36     log.debug("Received sign up request.")
37     (for (
38       authorizationHeader <- request.getAuthenticationHeader;
39       (username, password) <- HttpUtils.readBasicAuthentication(authorizationHeader)
40     ) yield {
41       storageFuture flatMap (_.signUpFuture(username, password)) onComplete {
42         case Success(_) =>
43           log.info(s"User $username signed up.")
44           JwtUtils
45             .encodeUsernameToken(username)
46             .foreach(sendResponse(CREATED, _))
47         case Failure(_) => sendResponse(BAD_REQUEST)
48       }
49     }) orElse Some(sendResponse(BAD_REQUEST))
50   }
51 
52   private def handlerSignOut: Handler[RoutingContext] = implicit routingContext => {
53     log.debug("Received sign out request.")
54     (for (
55       authorizationHeader <- request.getAuthenticationHeader;
56       token <- HttpUtils.readJwtAuthentication(authorizationHeader);
57       username <- JwtUtils.decodeUsernameToken(token)
58     ) yield {
59       // If every check pass, username contains the username contained in the token and we can check it exists
60       storageFuture.map(_.signOutFuture(username).onComplete {
61         case Success(_) =>
62           log.info(s"User $username signed out.")
63           sendResponse(ACCEPTED)
64         case Failure(_) => sendResponse(UNAUTHORIZED)
65       })
66     }) orElse Some(sendResponse(BAD_REQUEST))
67   }

76         case Success(_) =>

86   private def handlerValidation: Handler[RoutingContext] = implicit routingContext => {
87     log.debug("Received token validation request.")
88     (for (
89       authorizationHeader <- request.getAuthenticationHeader;
90       token <- HttpUtils.readJwtAuthentication(authorizationHeader);
91       username <- JwtUtils.decodeUsernameToken(token)
92     ) yield {
93       // If every check pass, username contains the username contained in the token and we can check it exists
94       storageFuture.map(_.existsFuture(username).onComplete {
95         case Success(_) =>
96           log.info(s"Token validation for $username successful")
97           sendResponse(OK, username)
98         case Failure(_) => sendResponse(UNAUTHORIZED)
99       })
100     }) orElse Some(sendResponse(BAD_REQUEST))
101   }
