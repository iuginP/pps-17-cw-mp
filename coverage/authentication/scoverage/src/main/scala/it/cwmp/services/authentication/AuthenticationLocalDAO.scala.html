67   def initialize(): Future[Unit] = {
68     log.info("Initializing RoomLocalDAO...")

77   override def signUpFuture(usernameP: String, passwordP: String): Future[Unit] = {
78     log.debug(s"signUp() username:$usernameP, password:$passwordP")
79     (for (
80       // Controllo l'input
81       username <- Option(usernameP) if username.nonEmpty;
82       password <- Option(passwordP) if password.nonEmpty
83     ) yield {
84       (for (
85         // Eseguo operazioni sul db in maniera sequenziale
86         _ <- checkInitialization(notInitialized);
87         connection <- openConnection();
88         _ <- connection.updateWithParamsFuture(
89           insertNewUserSql, Seq(username, password, "SALT"))
90       ) yield ()).closeConnections
91     }).getOrElse(Future.failed(new IllegalArgumentException(USERNAME_OR_PASSWORD_NOT_PROVIDED_ERROR)))
92   }
