65     *
66     * @return A Future that completes when initialization successful
67     */
68   def initialize(): Future[Unit] = {
69     log.info("Initializing RoomLocalDAO...")
70     (for (
71       connection <- openConnection();
72       _ <- connection.executeFuture(createStorageTableSql)
73     ) yield {
74       notInitialized = false
75     }).closeConnections
76   }

78   override def signUpFuture(usernameP: String, passwordP: String): Future[Unit] = {
79     log.debug(s"signUp() username:$usernameP, password:$passwordP")
80     (for (
81       // Controllo l'input
82       username <- Option(usernameP) if username.nonEmpty;
83       password <- Option(passwordP) if password.nonEmpty
84     ) yield {
85       (for (
86         // Eseguo operazioni sul db in maniera sequenziale
87         _ <- checkInitialization(notInitialized);
88         connection <- openConnection();
89         _ <- connection.updateWithParamsFuture(
90           insertNewUserSql, Seq(username, password, "SALT"))
91       ) yield ()).closeConnections
92     }).getOrElse(Future.failed(new IllegalArgumentException(USERNAME_OR_PASSWORD_NOT_PROVIDED_ERROR)))
93   }

108   }
