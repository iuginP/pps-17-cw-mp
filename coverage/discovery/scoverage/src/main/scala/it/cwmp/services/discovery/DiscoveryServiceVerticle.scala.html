20   * This verticle is composed by the web handler and the ServiceDiscovery.

49   private def handlerPublishService: Handler[RoutingContext] = implicit routingContext => {
50     log.debug("Received publish request.")
51     (for (
52       name <- request.getParam(PARAMETER_NAME);
53       host <- request.getParam(PARAMETER_HOST);
54       portString <- request.getParam(PARAMETER_PORT);
55       port = Integer.parseInt(portString);
56       requestRecord = HttpEndpoint.createRecord(name, host, port, "/")
57     ) yield {
58       //publish the service
59       discovery.publishFuture(requestRecord) onComplete {
60         case Success(record) =>
61           // Publication successful
62           log.info(s"Service ${record.getName} successfully published!")
63           sendResponse(CREATED, Some(record.asJava.toJson.toString))
64         case Failure(_) =>
65           // Publication failed
66           sendResponse(BAD_REQUEST)
67       }
68     }) orElse Some(sendResponse(BAD_REQUEST))
69   }
70 
71   private def handlerUnPublishService: Handler[RoutingContext] = implicit routingContext => {
72     log.debug("Received un-publish request.")
73     (for (
74       recordJson <- request.getParam(PARAMETER_REGISTRATION);
75       record <- try {
76         Some(Record.fromJson(new JsonObject(recordJson)))
77       } catch {
78         case _: Throwable => None
79       }
80     ) yield {
81       //publish the service
82       discovery.unpublishFuture(record.getRegistration) onComplete {
83         case Success(_) =>
84           // Un-publication successful
85           log.info(s"Service ${record.getName} successfully un-published!")
86           sendResponse(OK)
87         case Failure(_) =>
88           // Un-publication failed
89           sendResponse(BAD_REQUEST)
90       }
91     }) orElse Some(sendResponse(BAD_REQUEST))
92   }
93 
94   private def handlerDiscoverService: Handler[RoutingContext] = implicit routingContext => {
95     log.debug("Received discover request.")
96     (for (
97       name <- request.getParam(PARAMETER_NAME)
98     ) yield {
99       //publish the service
100       discovery.getRecordFuture(_.getName == name) onComplete {
101         case Success(record) if record.getStatus == Status.UP =>
102           // Publication successful
103           log.info(s"Service $name successfully found!")
104           val metadata: JsonObject = record.getLocation
105           sendResponse(OK, Some(metadata.toString))
106         case _ =>
107           // publication failed
108           sendResponse(BAD_REQUEST)
109       }
110     }) orElse Some(sendResponse(BAD_REQUEST))
111   }
