12 /**
13   * A trait describing the API wrapper for authentication service
14   */
15 trait DiscoveryApiWrapper {
16 
17   /**
18     * Publish the service in the remote discovery service.
19     *
20     * @param name the name of the service
21     * @param host the host on which the service can be reached
22     * @param port the port to use
23     * @return a successful Future if the service has been published, a failed Future otherwise.
24     */
25   def publish(name: String, host: String, port: Int): Future[Unit]
26 
27   /**
28     * Un-publish the service from the remote discovery service.
29     *
30     * @param name the name of the service
31     * @param host the host on which the service can be reached
32     * @param port the port to use
33     * @return a successful Future if the service has been successfully un-published, a failed Future otherwise.
34     */
35   def unPublish(name: String, host: String, port: Int): Future[Unit]
36 
37   /**
38     * This method is useful to search for the presence of a service
39     * of the specified type in the remote discovery service.
40     *
41     * @param name the name of the service to search
42     * @return a successful Future containing the couple (host, port) if a service is published, a failure otherwise.
43     */
44   def discover(name: String): Future[(String, Int)]
45 }

71     override def publish(name: String, host: String, port: Int): Future[Unit] = {
72       client.post(API_PUBLISH_SERVICE)
73         .addQueryParam(PARAMETER_NAME, name)
74         .addQueryParam(PARAMETER_HOST, host)
75         .addQueryParam(PARAMETER_PORT, port.toString)
76         .sendFuture()
77         .expectStatus(CREATED)
78         .map(_ => ())
79     }
80 
81     override def unPublish(name: String, host: String, port: Int): Future[Unit] = {
82       client.delete(API_UN_PUBLISH_SERVICE)
83         .addQueryParam(PARAMETER_NAME, name)
84         .addQueryParam(PARAMETER_HOST, host)
85         .addQueryParam(PARAMETER_PORT, port.toString)
86         .sendFuture()
87         .expectStatus(OK)
88         .map(_ => ())
89     }
90 
91     override def discover(name: String): Future[(String, Int)] = {
92       client.get(API_DISCOVER_SERVICE)
93         .addQueryParam(PARAMETER_NAME, name)
94         .sendFuture()
95         .expectStatus(OK)
96         .map(response => response
97           .bodyAsJsonObject()
98           .map(body => (body.getString("host"), body.getInteger("port").toInt))
99         )
100         // Filtering only the actual optional mapping the empty one into failures
101         .transform {
102           case s@Success(Some(_)) => s
103           case Success(None) => Failure(new Exception())
104           case f@Failure(_) => f
105         }
106         .map(_.get)
107     }
108   }
