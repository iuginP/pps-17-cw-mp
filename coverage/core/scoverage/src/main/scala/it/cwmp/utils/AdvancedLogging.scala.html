7 /**
8   * This trait provides a [[ch.qos.logback.classic.Logger]] as [[Logging]], but it has also a set of utility methods for logging.
9   *
10   * @author Eugenio Pierfederici
11   */
12 trait AdvancedLogging extends Logging {
13 
14   /**
15     * Enrich a future with logging capabilities.
16     *
17     * @param future the future to enrich.
18     * @tparam T the type of value provided from the future.
19     */
20   implicit class RichFuture[T](future: Future[T]) {
21 
22     /**
23       * Log an info message containing the text message if the future succeeds.
24       *
25       * @param message          the message to display.
26       * @param executionContext the implicit execution context on which to execute the operation.
27       * @return The future itself.
28       */
29     def logSuccessInfo(message: String)
30                       (implicit executionContext: ExecutionContext): Future[T] =
31       future.andThen {
32         case Success(_) => log.info(message)
33       }
34 
35     /**
36       * Log an info message containing the text message if the future succeeds and the condition is respected.
37       *
38       * @param message          the message to display.
39       * @param condition        the strategy that specifies if the argument of the successful future is right.
40       * @param executionContext the implicit execution context on which to execute the operation.
41       * @return The future itself.
42       */
43     def logSuccessInfo(message: String, condition: T => Boolean)
44                       (implicit executionContext: ExecutionContext): Future[T] =
45       future.andThen {
46         case Success(s) if condition(s) => log.info(message)
47       }
48 
49     /**
50       * Log an info message containing the text message if the future fails.
51       *
52       * @param message          the message to display.
53       * @param executionContext the implicit execution context on which to execute the operation.
54       * @return The future itself.
55       */
56     def logFailureInfo(message: String)
57                       (implicit executionContext: ExecutionContext): Future[T] =
58       future.andThen {
59         case Failure(e) => log.info(message, e)
60       }
61 
62     /**
63       * Log an info message containing the text message if the future fails with the specified exception type.
64       *
65       * @param message the message to display.
66       * @tparam A the type of exception that should be returned as failure.
67       * @param executionContext the implicit execution context on which to execute the operation.
68       * @return The future itself.
69       */
70     def logFailureInfo[A <: Exception : ClassTag](message: String)
71                                                  (implicit executionContext: ExecutionContext): Future[T] =
72       future.andThen {
73         case Failure(e) if classTag[A].runtimeClass.isInstance(e) => log.info(message, e)
74       }
75   }
