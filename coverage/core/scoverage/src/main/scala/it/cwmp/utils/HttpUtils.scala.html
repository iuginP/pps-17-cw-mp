16   /**
17     * Build the basic authentication header for the http request.
18     * It si composed from the [[PREFIX_BASIC]] prefix followed by the Base64 of 'username:password'
19     *
20     * @param username the username
21     * @param password the password
22     * @return the header for the request, or None
23     */
24   def buildBasicAuthentication(username: String, password: String): Option[String] =
25     for (
26       usernameVal <- Option(username) if usernameVal.nonEmpty;
27       passwordVal <- Option(password) if passwordVal.nonEmpty
28     ) yield s"$PREFIX_BASIC " + Base64.getEncoder.encodeToString(s"$usernameVal:$passwordVal".getBytes())
29 
30   /**
31     * Reads the couple (username,password) from the authentication header in the request.
32     *
33     * @param header the authentication header
34     * @return the couple containing username and password, or None
35     */
36   def readBasicAuthentication(header: String): Option[(String, String)] =
37     try {
38       for (
39         headerVal <- Option(header) if headerVal.nonEmpty;
40         //divido la stringa "Basic " dalla parte in Base64
41         credentialFromHeader = headerVal.split(s"$PREFIX_BASIC ")(1) if credentialFromHeader.nonEmpty;
42         //decodifico il payload e ottengo, quando Ã¨ possibile, username e password in chiaro
43         headerDecoded = new String(Base64.getDecoder.decode(credentialFromHeader)).split(":")
44       ) yield (headerDecoded(0), headerDecoded(1))
45     } catch {
46       case _: Throwable => None
47     }
48 
49   /**
50     * Build the jwt authentication header for the http request.
51     * It si composed from the [[PREFIX_JWT]] prefix followed by the real JWT token.
52     *
53     * @param token the token
54     * @return the header for the request, or None
55     */
56   def buildJwtAuthentication(token: String): Option[String] =
57     for (
58       tokenVal <- Option(token) if tokenVal.nonEmpty
59     ) yield s"$PREFIX_JWT $tokenVal"

61   /**

63     *
64     * @param header the authentication header

67   def readJwtAuthentication(header: String): Option[String] =
68     try {
69       for (
70         headerVal <- Option(header) if headerVal.nonEmpty;
71         token = headerVal.split(s"$PREFIX_JWT ")(1) if token.nonEmpty

74       case _: Throwable => None
