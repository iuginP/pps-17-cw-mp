13   * This trait provides some utilities for an easier management of the JDBC connection with vertx.
14   *
15   * @author Eugenio Pierfederici

22   private val DEFAULT_CONFIG_PATH: String = "database/jdbc_config.json"

24   private val clientFuture: Future[JDBCClient] =

30   private var connectionList: ListBuffer[SQLConnection] = ListBuffer()
31 
32   /**
33     * Open a connection and returns it.
34     *
35     * @return the connection just opened.
36     */
37   protected def openConnection(): Future[SQLConnection] = {
38     clientFuture.flatMap(_.getConnectionFuture()).andThen {
39       case Success(connection) => connectionList += connection
40     }
41   }
42 
43   /**
44     * Closes the specified connection (or present as implicit)
45     *
46     * @param connection the connection we want to close.
47     */
48   protected def closeConnection(implicit connection: SQLConnection): Unit = {
49     connectionList -= connection
50     connection.close()
51   }
52 
53   /**
54     * Closes all the connections opened until now through this client.
55     */
56   protected def closeAllConnections(): Unit = connectionList.foreach(closeConnection(_))
57 
58   /**
59     * Closes last opened connection through this client.
60     */
61   protected def closeLastOpenedConnection(): Unit =
62     if (connectionList.nonEmpty) closeConnection(connectionList.last)
63 
64   /**
65     * This class decorates the future with some utils for the connection management.
66     *
67     * @param future the future to decorate
68     * @tparam F the type of the future
69     */
70   implicit class RichFuture[F](future: Future[F]) {
71 
72     /**
73       * When the future reaches that point, it closes all the connections opened until now in the client.
74       *
75       * @param executionContext the execution context in which to execute the operation
76       * @return the future itself
77       */
78     def closeConnections(implicit executionContext: ExecutionContext): Future[F] =
79       future.andThen { case _ => closeAllConnections() }
80 
81     /**
82       * When the future reches this point, it closes the last opened connection till now in the client.
83       *
84       * @param executionContext the execution context in which to execute the operation
85       * @return the future itself
86       */
87     def closeLastConnection(implicit executionContext: ExecutionContext): Future[F] =
88       future.andThen { case _ => closeLastOpenedConnection() }
89   }
