13 /**
14   * A trait describing the API wrapper for authentication service
15   */
16 trait AuthenticationApiWrapper extends Validation[String, User] {
17 
18   /**
19     * It tries to execute a sign up on the system. The implementation specifies how the operation should be executed.
20     *
21     * @param username the username to register
22     * @param password the password for that user
23     * @return a future that may be satisfied when the operation completes successfully, otherwise it fails.
24     */
25   def signUp(username: String, password: String): Future[String]
26 
27   /**
28     * Allow a user to delete his account and sign out from the system.
29     *
30     * @param token the authentication token
31     * @return a future that succeed only if the user has been successfully deleted.
32     */
33   def signOut(token: String): Future[Unit]
34 
35   /**
36     * It tries to execute a login on the system. The implementation specifies how the operation should be executed.
37     *
38     * @param username the username of the user that is trying to authenticate itself
39     * @param password the password for that user
40     * @return a future that may be satisfied when the operation completes successfully, otherwise it fails.
41     */
42   def login(username: String, password: String): Future[String]
43 }

77       client.delete(API_SIGN_OUT)

83     override def login(username: String, password: String): Future[String] =
84       client.get(API_LOGIN)
85         .addAuthentication(username, password)
86         .sendFuture()
87         .expectStatus(OK)
88         .map(_.bodyAsString().getOrElse(""))
89 
90     override def validate(authenticationHeader: String): Future[User] =
91       client.get(API_VALIDATE)
92         .addAuthenticationHeader(authenticationHeader)
93         .sendFuture()
94         .expectStatus(OK)
95         .mapBody {
96           case Some(body) => Future.successful(User(body))
97           case _ => Future.failed(HTTPException(BAD_REQUEST, "Empty body"))
98         }
