12 /**
13   * This trait provides some utilities for an easier management of the JDBC connection with vertx.
14   *
15   * @author Eugenio Pierfederici
16   */
17 trait VertxJDBC {
18   this: VertxInstance =>
19 
20   def configurationPath: Option[String] = None
21 
22   private val DEFAULT_CONFIG_PATH: String = "database/jdbc_config.json"
23 
24   private val clientFuture: Future[JDBCClient] =
25     for (
26       configPath <- Future(configurationPath.getOrElse(DEFAULT_CONFIG_PATH));
27       configExists <- vertx.fileSystem.existsFuture(configPath);
28       configFileBuffer <- vertx.fileSystem.readFileFuture(if (configExists) configPath else DEFAULT_CONFIG_PATH)
29     ) yield JDBCClient.createShared(vertx, configFileBuffer.toJsonObject)
30 
31   private var connectionList: ListBuffer[SQLConnection] = ListBuffer()
32 
33   /**
34     * Open a connection and returns it.
35     *
36     * @return the connection just opened.
37     */
38   protected def openConnection(): Future[SQLConnection] = {
39     clientFuture.flatMap(_.getConnectionFuture()).andThen {
40       case Success(connection) => connectionList += connection
41     }
42   }
43 
44   /**
45     * Closes the specified connection (or present as implicit)
46     *
47     * @param connection the connection we want to close.
48     */
49   protected def closeConnection(implicit connection: SQLConnection): Unit = {
50     connectionList -= connection
51     connection.close()
52   }
53 
54   /**
55     * Closes all the connections opened until now through this client.
56     */
57   protected def closeAllConnections(): Unit = connectionList.foreach(closeConnection(_))
58 
59   /**
60     * Closes last opened connection through this client.
61     */
62   protected def closeLastOpenedConnection(): Unit =
63     if (connectionList.nonEmpty) closeConnection(connectionList.last)
64 
65   /**
66     * This class decorates the future with some utils for the connection management.
67     *
68     * @param future the future to decorate
69     * @tparam F the type of the future
70     */
71   implicit class RichFuture[F](future: Future[F]) {
72 
73     /**
74       * When the future reaches that point, it closes all the connections opened until now in the client.
75       *
76       * @param executionContext the execution context in which to execute the operation
77       * @return the future itself
78       */
79     def closeConnections(implicit executionContext: ExecutionContext): Future[F] =
80       future.andThen { case _ => closeAllConnections() }
81 
82     /**
83       * When the future reaches this point, it closes the last opened connection till now in the client.
84       *
85       * @param executionContext the execution context in which to execute the operation
86       * @return the future itself
87       */
88     def closeLastConnection(implicit executionContext: ExecutionContext): Future[F] =
89       future.andThen { case _ => closeLastOpenedConnection() }
90   }
91 
92 }
