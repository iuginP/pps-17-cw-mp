27   override def startFuture(): Future[_] = {
28     val router = Router.router(vertx)
29     initRouter(router)
30     initServer.flatMap(_ =>
31       vertx.createHttpServer()
32         .requestHandler(router.accept _)
33         .listenFuture(serverPort))
34       .andThen {
35         case Success(s) =>

39       }
40   }

108     /**
109       * This is a utility method that checks if the request is authenticated.
110       * For the check it uses the validation strategy given.
111       *
112       * @param strategy       the validation strategy to use.
113       * @param routingContext the routing context in which to execute the check.
114       * @return A future containing the authenticated user, if present, otherwise it fails with a [[HTTPException]]
115       */
116     def checkAuthentication(implicit strategy: Validation[String, User],
117                             routingContext: RoutingContext): Future[User] = getAuthenticationHeader match {
118       case None =>
119         log.warn(NO_AUTH_HEADER_IN_REQUEST_ERROR)
120         Future.failed(HTTPException(BAD_REQUEST, NO_AUTH_HEADER_IN_REQUEST_ERROR))
121       case Some(authenticationHeader) => strategy.validate(authenticationHeader)
122     }
123 
124     /**
125       * This is a utility method that checks if the request is authenticated.
126       * For the check it uses the validation strategy given.
127       * If it fails with any [[HTTPException]], then it responds to the request
128       * with the status code and message specified in the exception.
129       *
130       * @param strategy       the validation strategy to use.
131       * @param routingContext the routing context in which to execute the check.
132       * @return A future containing the authenticated user, if present
133       */
134     def checkAuthenticationOrReject(implicit strategy: Validation[String, User],
135                                     routingContext: RoutingContext): Future[User] =
136       checkAuthentication(strategy, routingContext).recoverWith {
137         case HTTPException(statusCode, errorMessage) =>
138           sendResponse(statusCode, errorMessage)
139           Future.failed(new IllegalAccessException(errorMessage))
140       }
141 
142     /**
