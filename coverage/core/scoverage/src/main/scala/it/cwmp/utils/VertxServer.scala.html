25   override def startFuture(): Future[_] = {
26     val router = Router.router(vertx)
27     initRouter(router)
28     initServer.flatMap(_ =>
29       vertx.createHttpServer()
30         .requestHandler(router.accept _)
31         .listenFuture(serverPort))
32       .andThen {
33         case Success(s) =>

37       }
38   }

109     /**

111       * For the check it uses the validation strategy given.
112       *
113       * @param strategy       the validation strategy to use.

116       */
117     def checkAuthentication(implicit strategy: Validation[String, User],
118                             routingContext: RoutingContext): Future[User] = getAuthenticationHeader match {
119       case None =>
120         log.warn(NO_AUTH_HEADER_IN_REQUEST_ERROR)
121         Future.failed(HTTPException(400, NO_AUTH_HEADER_IN_REQUEST_ERROR))
122       case Some(authenticationHeader) => strategy.validate(authenticationHeader)
123     }

125     /**
126       * This is a utility method that checks if the request is authenticated.
127       * For the check it uses the validation strategy given.
128       * If it fails with any [[HTTPException]], then it responds to the request
129       * with the status code and message specified in the exception.
130       *
131       * @param strategy       the validation strategy to use.
132       * @param routingContext the routing context in which to execute the check.
133       * @return A future containing the authenticated user, if present
134       */
135     def checkAuthenticationOrReject(implicit strategy: Validation[String, User],
136                                     routingContext: RoutingContext): Future[User] =
137       checkAuthentication(strategy, routingContext).recoverWith {
138         case HTTPException(statusCode, errorMessage) =>
139           sendResponse(statusCode, errorMessage)
140           Future.failed(new IllegalAccessException(errorMessage))
141       }
142 
143     /**
144       * Reads the authorization header (with token) in the request.
145       *
146       * @return An optional containing the header, if present. Otherwise None
147       */
148     def getAuthenticationHeader: Option[String] = request.getHeader(HttpHeaderNames.AUTHORIZATION.toString)
149   }
