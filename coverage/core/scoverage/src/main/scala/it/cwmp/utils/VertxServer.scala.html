26   override def startFuture(): Future[_] = {
27     val router = Router.router(vertx)
28     initRouter(router)

120       case None =>

122         Future.failed(HTTPException(BAD_REQUEST, NO_AUTH_HEADER_IN_REQUEST_ERROR))
123       case Some(authenticationHeader) => strategy.validate(authenticationHeader)
124     }

126     /**

128       * For the check it uses the validation strategy given.
129       * If it fails with any [[HTTPException]], then it responds to the request
130       * with the status code and message specified in the exception.
131       *
132       * @param strategy       the validation strategy to use.
133       * @param routingContext the routing context in which to execute the check.
134       * @return A future containing the authenticated user, if present
135       */
136     def checkAuthenticationOrReject(implicit strategy: Validation[String, User],
137                                     routingContext: RoutingContext): Future[User] =
138       checkAuthentication(strategy, routingContext).recoverWith {
139         case HTTPException(statusCode, errorMessage) =>
140           sendResponse(statusCode, errorMessage)
141           Future.failed(new IllegalAccessException(errorMessage))
142       }
143 
144     /**
