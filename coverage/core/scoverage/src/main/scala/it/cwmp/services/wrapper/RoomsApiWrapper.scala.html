141   private class RoomsApiWrapperDefault(override protected val clientOptions: WebClientOptions)

143 
144     override def createRoom(roomName: String, playersNumber: Int)
145                            (implicit userToken: String): Future[String] =
146       createPrivateRoomRequest(roomName, playersNumber)
147         .flatMap(implicit response => handleResponse(Future.successful(response.bodyAsString().get), 201))
148 
149     override def enterRoom(roomID: String, userAddress: Address, notificationAddress: Address)
150                           (implicit userToken: String): Future[Unit] =
151       enterPrivateRoomRequest(roomID, userAddress, notificationAddress)
152         .flatMap(implicit response => handleResponse(Future.successful(()), 200))
153 
154     override def roomInfo(roomID: String)
155                          (implicit userToken: String): Future[Room] =
156       privateRoomInfoRequest(roomID)
157         .flatMap(implicit response =>
158           handleResponse(Future.successful(Json.fromObjectString(response.bodyAsString().get).toRoom), 200))
159 
160     override def exitRoom(roomID: String)
161                          (implicit userToken: String): Future[Unit] =
162       exitPrivateRoomRequest(roomID)
163         .flatMap(implicit response => handleResponse(Future.successful(()), 200))
164 
165     override def listPublicRooms()(implicit userToken: String): Future[Seq[Room]] =
166       listPublicRoomsRequest()
167         .flatMap(implicit response => handleResponse({
168           Future.successful {
169             Json.fromArrayString(response.bodyAsString().get)
170               .stream().toArray().toSeq.map(_.toString)
171               .map(jsonString => Json.fromObjectString(jsonString).toRoom)
172           }
173         }, 200))
174 
175     override def enterPublicRoom(playersNumber: Int, userAddress: Address, notificationAddress: Address)
176                                 (implicit userToken: String): Future[Unit] =
177       enterPublicRoomRequest(playersNumber, userAddress, notificationAddress)
178         .flatMap(implicit response => handleResponse(Future.successful(()), 200))
179 
180     override def publicRoomInfo(playersNumber: Int)
181                                (implicit userToken: String): Future[Room] =
182       publicRoomInfoRequest(playersNumber)
183         .flatMap(implicit response =>
184           handleResponse(Future.successful(Json.fromObjectString(response.bodyAsString().get).toRoom), 200))
185 
186     override def exitPublicRoom(playersNumber: Int)
187                                (implicit userToken: String): Future[Unit] =
188       exitPublicRoomRequest(playersNumber)
189         .flatMap(implicit response => handleResponse(Future.successful(()), 200))
190 
191     /**
192       * Utility method to handle the Service response
193       */
194     private def handleResponse[T](onSuccessFuture: => Future[T], successHttpCodes: Int*)
195                                  (implicit response: HttpResponse[Buffer]) =
196       successHttpCodes find (_ == response.statusCode) match {
197         case Some(_) => onSuccessFuture
198         case None => response.bodyAsString match {
199           case Some(body) => Future.failed(HTTPException(response.statusCode, body))
200           case None => Future.failed(HTTPException(response.statusCode))
201         }
202       }
203   }
