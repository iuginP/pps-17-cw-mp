25   /**

37   /**
38     * Enters a room
39     *
40     * @param roomID              the identifier of the room
41     * @param userAddress         the address of the paler that wants to enter
42     * @param notificationAddress the address where player wants to receive notification of room filling
43     * @param userToken           the token to be authenticated against api
44     * @return the future that completes when the user has entered,
45     *         or fails if roomID not provided, not present
46     *         or user already inside a room, or room full

48   def enterRoom(roomID: String, userAddress: Address, notificationAddress: Address)

59   def roomInfo(roomID: String)

146     override def createRoom(roomName: String, playersNumber: Int)
147                            (implicit userToken: String): Future[String] =

149         .flatMap(implicit response => handleResponse(Future.successful(response.bodyAsString().get), CREATED))
150 
151     override def enterRoom(roomID: String, userAddress: Address, notificationAddress: Address)
152                           (implicit userToken: String): Future[Unit] =

154         .flatMap(implicit response => handleResponse(Future.successful(()), OK))
155 
156     override def roomInfo(roomID: String)
157                          (implicit userToken: String): Future[Room] =
158       privateRoomInfoRequest(roomID)
159         .flatMap(implicit response =>
160           handleResponse(Future.successful(Json.fromObjectString(response.bodyAsString().get).toRoom), OK))
161 
162     override def exitRoom(roomID: String)
163                          (implicit userToken: String): Future[Unit] =
164       exitPrivateRoomRequest(roomID)
165         .flatMap(implicit response => handleResponse(Future.successful(()), OK))
166 
167     override def listPublicRooms()(implicit userToken: String): Future[Seq[Room]] =
168       listPublicRoomsRequest()
169         .flatMap(implicit response => handleResponse({
170           Future.successful {
171             Json.fromArrayString(response.bodyAsString().get)
172               .stream().toArray().toSeq.map(_.toString)
173               .map(jsonString => Json.fromObjectString(jsonString).toRoom)
174           }
175         }, OK))
176 
177     override def enterPublicRoom(playersNumber: Int, userAddress: Address, notificationAddress: Address)
178                                 (implicit userToken: String): Future[Unit] =
179       enterPublicRoomRequest(playersNumber, userAddress, notificationAddress)
180         .flatMap(implicit response => handleResponse(Future.successful(()), OK))
181 
182     override def publicRoomInfo(playersNumber: Int)
183                                (implicit userToken: String): Future[Room] =
184       publicRoomInfoRequest(playersNumber)
185         .flatMap(implicit response =>
186           handleResponse(Future.successful(Json.fromObjectString(response.bodyAsString().get).toRoom), OK))
187 
188     override def exitPublicRoom(playersNumber: Int)
189                                (implicit userToken: String): Future[Unit] =
190       exitPublicRoomRequest(playersNumber)
191         .flatMap(implicit response => handleResponse(Future.successful(()), OK))
192 
193     /**
194       * Utility method to handle the Service response
195       */
196     private def handleResponse[T](onSuccessFuture: => Future[T], successHttpCodes: Int*)
197                                  (implicit response: HttpResponse[Buffer]) =
198       successHttpCodes find (_ == response.statusCode) match {
199         case Some(_) => onSuccessFuture
200         case None => response.bodyAsString match {
201           case Some(body) => Future.failed(HTTPException(response.statusCode, body))
202           case None => Future.failed(HTTPException(response.statusCode))
203         }
204       }
205   }
