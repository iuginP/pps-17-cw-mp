68       try {
69         val jsonObject = body.toJsonObject

72         } else {
73           None
74         }
75       } catch {
76         case _: Throwable => None
77       }

239   /**
240     * Describes how to behave when a room is filled out
241     *
242     * @param roomInformationFuture the future that will contain the room and addresses of users to contact
243     * @param onRetrievedRoomAction the action to do if the room is full
244     * @return a future that completes when all players received addresses
245     */
246   private def handleRoomFilling(roomInformationFuture: Future[(Room, Seq[Address])], onRetrievedRoomAction: => Future[Unit])
247                                (implicit communicationStrategy: RoomReceiverApiWrapper, executionContext: ExecutionContext): Future[Unit] = {
248     roomInformationFuture
249       .map(roomAndAddresses => roomAndAddresses._1)
250       .filter(roomIsFull)
251       .flatMap(_ => {
252         onRetrievedRoomAction
253         sendParticipantAddresses(roomInformationFuture.value.get.get)(communicationStrategy, executionContext)
254       }).transform {
255       case Success(_) => Failure(new Exception()) // if operation successful, outer response sending should block
256       case Failure(_: NoSuchElementException) => Success(()) // if room wasn't full, let outer response sending happen
257       case Failure(ex) => log.error("Error handling Room Filling", ex); Success(())
258       // unexpected error, log and let outer response sending happen
259     }
260   }

262   /**
263     * Method to communicate to clients that the game can start because of reaching the specified number of players
264     *
265     * @param roomInformation the room where players are waiting in
266     */
267   private def sendParticipantAddresses(roomInformation: (Room, Seq[Address]))
268                                       (implicit communicationStrategy: RoomReceiverApiWrapper, executionContext: ExecutionContext): Future[Unit] = {
269     log.info(s"Preparing to send participant list to room ${roomInformation._1.name} (with id:${roomInformation._1.identifier}) participants ...")
270     val notificationAddresses = for (notificationAddress <- roomInformation._2) yield notificationAddress
271     val players = for (player <- roomInformation._1.participants) yield player
272     log.info(s"Participant notification addresses to contact: $notificationAddresses")
273     log.info(s"Information to send -> $players")
274     Future.sequence {
275       notificationAddresses map (notificationAddress => communicationStrategy.sendParticipants(notificationAddress.address, players))
276     } map (_ => log.info("Information sent."))
277   }
