54   private def createPrivateRoomHandler: Handler[RoutingContext] = implicit routingContext => {
55     log.info("Create Private Room Request received...")
56     request.bodyHandler(body =>
57       request.checkAuthenticationOrReject.map(_ =>
58         extractIncomingRoomFromBody(body) match {
59           case Some((roomName, neededPlayers)) =>
60             daoFuture.map(_.createRoom(roomName, neededPlayers).onComplete {
61               failureHandler orElse successHandler(generatedID => sendResponse(CREATED, generatedID))
62             })
63           case None =>
64             log.warn("Request body for room creation is required!")
65             sendResponse(BAD_REQUEST, s"$INVALID_PARAMETER_ERROR no Room JSON in body")
66         }))
67 
68     def extractIncomingRoomFromBody(body: Buffer): Option[(String, Int)] = {
69       try {
70         val jsonObject = body.toJsonObject
71         if ((jsonObject containsKey Room.FIELD_NAME) && (jsonObject containsKey Room.FIELD_NEEDED_PLAYERS)) {
72           Some((jsonObject getString Room.FIELD_NAME, jsonObject getInteger Room.FIELD_NEEDED_PLAYERS))
73         } else {
74           None
75         }
76       } catch {
77         case _: Throwable => None
78       }
79     }

240   /**
241     * Describes how to behave when a room is filled out
242     *
243     * @param roomInformationFuture the future that will contain the room and addresses of users to contact
244     * @param onRetrievedRoomAction the action to do if the room is full
245     * @return a future that completes when all players received addresses
246     */
247   private def handleRoomFilling(roomInformationFuture: Future[(Room, Seq[Address])], onRetrievedRoomAction: => Future[Unit])
248                                (implicit communicationStrategy: RoomReceiverApiWrapper, executionContext: ExecutionContext): Future[Unit] = {
249     roomInformationFuture
250       .map(roomAndAddresses => roomAndAddresses._1)
251       .filter(roomIsFull)
252       .flatMap(_ => {
253         onRetrievedRoomAction
254         sendParticipantAddresses(roomInformationFuture.value.get.get)(communicationStrategy, executionContext)
255       }).transform {
256       case Success(_) => Failure(new Exception()) // if operation successful, outer response sending should block
257       case Failure(_: NoSuchElementException) => Success(()) // if room wasn't full, let outer response sending happen
258       case Failure(ex) => log.error("Error handling Room Filling", ex); Success(())
259       // unexpected error, log and let outer response sending happen
260     }
261   }

263   /**
264     * Method to communicate to clients that the game can start because of reaching the specified number of players
265     *
266     * @param roomInformation the room where players are waiting in
267     */
268   private def sendParticipantAddresses(roomInformation: (Room, Seq[Address]))
269                                       (implicit communicationStrategy: RoomReceiverApiWrapper, executionContext: ExecutionContext): Future[Unit] = {
270     log.info(s"Preparing to send participant list to room ${roomInformation._1.name} (with id:${roomInformation._1.identifier}) participants ...")
271     val notificationAddresses = for (notificationAddress <- roomInformation._2) yield notificationAddress
272     val players = for (player <- roomInformation._1.participants) yield player
273     log.info(s"Participant notification addresses to contact: $notificationAddresses")
274     log.info(s"Information to send -> $players")
275     Future.sequence {
276       notificationAddresses map (notificationAddress => communicationStrategy.sendParticipants(notificationAddress.address, players))
277     } map (_ => log.info("Information sent."))
278   }
