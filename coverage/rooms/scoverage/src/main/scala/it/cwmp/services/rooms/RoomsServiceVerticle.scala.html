64             sendResponse(400, s"$INVALID_PARAMETER_ERROR no Room JSON in body")

68       try {

74         case _: Throwable => None

237   /**
238     * Describes how to behave when a room is filled out
239     *
240     * @param roomInformationFuture the future that will contain the room and addresses of users to contact
241     * @param onRetrievedRoomAction the action to do if the room is full
242     * @return a future that completes when all players received addresses
243     */
244   private def handleRoomFilling(roomInformationFuture: Future[(Room, Seq[Address])], onRetrievedRoomAction: => Future[Unit])
245                                (implicit communicationStrategy: RoomReceiverApiWrapper, executionContext: ExecutionContext): Future[Unit] = {
246     roomInformationFuture
247       .map(roomAndAddresses => roomAndAddresses._1)
248       .filter(roomIsFull)
249       .flatMap(_ => {
250         onRetrievedRoomAction
251         sendParticipantAddresses(roomInformationFuture.value.get.get)(communicationStrategy, executionContext)
252       }).transform {
253       case Success(_) => Failure(new Exception()) // if operation successful, outer response sending should block
254       case Failure(_: NoSuchElementException) => Success(()) // if room wasn't full, let outer response sending happen
255       case Failure(ex) => log.error("Error handling Room Filling", ex); Success(())
256       // unexpected error, log and let outer response sending happen
257     }
258   }

260   /**
261     * Method to communicate to clients that the game can start because of reaching the specified number of players
262     *
263     * @param roomInformation the room where players are waiting in
264     */
265   private def sendParticipantAddresses(roomInformation: (Room, Seq[Address]))
266                                       (implicit communicationStrategy: RoomReceiverApiWrapper, executionContext: ExecutionContext): Future[Unit] = {
267     log.info(s"Preparing to send participant list to room ${roomInformation._1.name} (with id:${roomInformation._1.identifier}) participants ...")
268     val notificationAddresses = for (notificationAddress <- roomInformation._2) yield notificationAddress
269     val players = for (player <- roomInformation._1.participants) yield player
270     log.info(s"Participant notification addresses to contact: $notificationAddresses")
271     log.info(s"Information to send -> $players")
272     Future.sequence {
273       notificationAddresses map (notificationAddress => communicationStrategy.sendParticipants(notificationAddress.address, players))
274     } map (_ => log.info("Information sent."))
275   }
