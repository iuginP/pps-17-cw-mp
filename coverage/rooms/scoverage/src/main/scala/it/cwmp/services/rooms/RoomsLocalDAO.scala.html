274   override def deleteRoom(roomID: String): Future[Unit] = {
275 
276     def giveErrorOnNonFullRoom(roomID: String, connection: SQLConnection): Future[Unit] =
277       checkRoomSpaceAvailable(roomID, connection, "") // if no room space available, will give empty error message
278         // if no error received, should fail -> because room non full
279         .flatMap(_ => Future.failed(new IllegalStateException(DELETING_NON_FULL_ROOM_ERROR)))
280         // if first error reaches this point, room is full and can be deleted
281         .recoverWith({ case ex: IllegalStateException if ex.getMessage.isEmpty => Future.successful(()) })
282 
283     log.debug(s"deleteRoom() roomID:$roomID")
284     (for (
285       _ <- checkInitialization(notInitialized);
286       room <- roomInfo(roomID).map(_._1);
287       conn <- openConnection();
288       _ <- giveErrorOnNonFullRoom(roomID, conn);
289 
290       // outer for will reach this point only if room is full (and can be deleted)
291       userDeletionFutures = for (user <- room.participants;
292                                  deletionFuture = conn.updateWithParamsFuture(deleteUserFormRoomSql, Seq(user.username))) yield deletionFuture;
293       _ <- Future.sequence(userDeletionFutures); // waits for all deletions to complete
294       _ <- conn.updateWithParamsFuture(deleteRoomSql, Seq(roomID))
295     ) yield ()).closeLastConnection
296   }

509     /**
510       * Extracts a map of roomIds to Room participant list from the resultSet passed in
511       *
512       * @param resultSet      the result set to use extracting information
513       * @param roomIDPos      the position of roomID in result rows
514       * @param userNamePos    the position of userName in result rows

516       * @return the map that has roomId as key and room participants as value

518     private def roomsAndUsersInside(resultSet: ResultSet,
519                                     roomIDPos: Int,
520                                     userNamePos: Int,
521                                     userAddressPos: Int): mutable.Map[String, Seq[Participant]] = {
522       val roomsUsers: mutable.Map[String, Seq[Participant]] = mutable.HashMap()
523       for (resultRow <- resultSet.getResults) {
524         val roomID = resultRow getString roomIDPos
525         val userName = if (resultRow hasNull userNamePos) None else Some(resultRow getString userNamePos)
526         val userAddress = if (resultRow hasNull userAddressPos) None else Some(resultRow getString userAddressPos)
527 
528         if (roomsUsers contains roomID) {
529           userName.foreach(name => roomsUsers(roomID) = roomsUsers(roomID) :+ Participant(name, userAddress.get))
530         } else {
531           (userName, userAddress) match {
532             case (Some(name), Some(address)) => roomsUsers(roomID) = Seq(Participant(name, address))
533             case _ => roomsUsers(roomID) = Seq()
534           }
535         }
536       }
537       roomsUsers
