275   override def deleteRoom(roomID: String): Future[Unit] = {
276 
277     def giveErrorOnNonFullRoom(roomID: String, connection: SQLConnection): Future[Unit] =
278       checkRoomSpaceAvailable(roomID, connection, "") // if no room space available, will give empty error message
279         // if no error received, should fail -> because room non full
280         .flatMap(_ => Future.failed(new IllegalStateException(DELETING_NON_FULL_ROOM_ERROR)))
281         // if first error reaches this point, room is full and can be deleted
282         .recoverWith({ case ex: IllegalStateException if ex.getMessage.isEmpty => Future.successful(()) })
283 
284     log.debug(s"deleteRoom() roomID:$roomID")
285     (for (
286       _ <- checkInitialization(notInitialized);
287       room <- roomInfo(roomID).map(_._1);
288       conn <- openConnection();
289       _ <- giveErrorOnNonFullRoom(roomID, conn);
290 
291       // outer for will reach this point only if room is full (and can be deleted)
292       userDeletionFutures = for (user <- room.participants;
293                                  deletionFuture = conn.updateWithParamsFuture(deleteUserFormRoomSql, Seq(user.username))) yield deletionFuture;
294       _ <- Future.sequence(userDeletionFutures); // waits for all deletions to complete
295       _ <- conn.updateWithParamsFuture(deleteRoomSql, Seq(roomID))
296     ) yield ()).closeLastConnection

511       * Extracts a map of roomIds to Room participant list from the resultSet passed in
512       *
513       * @param resultSet      the result set to use extracting information
514       * @param roomIDPos      the position of roomID in result rows
515       * @param userNamePos    the position of userName in result rows

522                                     userAddressPos: Int): mutable.Map[String, Seq[Participant]] = {
523       val roomsUsers: mutable.Map[String, Seq[Participant]] = mutable.HashMap()
524       for (resultRow <- resultSet.getResults) {
525         val roomID = resultRow getString roomIDPos
526         val userName = if (resultRow hasNull userNamePos) None else Some(resultRow getString userNamePos)
527         val userAddress = if (resultRow hasNull userAddressPos) None else Some(resultRow getString userAddressPos)
528 
529         if (roomsUsers contains roomID) {
530           userName.foreach(name => roomsUsers(roomID) = roomsUsers(roomID) :+ Participant(name, userAddress.get))
531         } else {
532           (userName, userAddress) match {
533             case (Some(name), Some(address)) => roomsUsers(roomID) = Seq(Participant(name, address))
534             case _ => roomsUsers(roomID) = Seq()
535           }

537       }
