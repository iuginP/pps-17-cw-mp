272 
273     def giveErrorOnNonFullRoom(roomID: String, connection: SQLConnection): Future[Unit] =
274       checkRoomSpaceAvailable(roomID, connection, "") // if no room space available, will give empty error message

285       _ <- giveErrorOnNonFullRoom(roomID, conn);
286 
287       // outer for will reach this point only if room is full (and can be deleted)

398   /**
399     * Utility method to convert a result set to a room sequence and relative addresses
400     */
401   private def resultOfJoinToRooms(resultSet: ResultSet): Seq[(Room, Seq[Address])] = {
402     val roomsUsers: mutable.Map[String, Seq[Participant]] = mutable.HashMap()
403     val roomsAddresses: mutable.Map[String, Seq[Address]] = mutable.HashMap()
404     val roomsInfo: mutable.Map[String, (String, Int)] = mutable.HashMap()
405     val roomIDPos = 0
406     val roomNamePos = 1
407     val roomPlayersPos = 2
408     val userNamePos = 3
409     val userAddressPos = 4
410     val userNotificationAddressPos = 5
411 
412     for (resultRow <- resultSet.getResults) {
413       val roomID = resultRow getString roomIDPos
414       val userName = if (resultRow hasNull userNamePos) None else Some(resultRow getString userNamePos)
415       val userAddress = if (resultRow hasNull userAddressPos) None else Some(resultRow getString userAddressPos)
416       val userNotificationAddress = if (resultRow hasNull userNotificationAddressPos) None else Some(resultRow getString userNotificationAddressPos)
417 
418       if (roomsUsers contains roomID) {
419         userName.foreach(name => roomsUsers(roomID) = roomsUsers(roomID) :+ Participant(name, userAddress.get))
420         userNotificationAddress.foreach(notificationAddress => roomsAddresses(roomID) = roomsAddresses(roomID) :+ Address(notificationAddress))
421       } else {
422         val roomName = resultRow getString roomNamePos
423         val roomNeededPlayers = resultRow getInteger roomPlayersPos
424         roomsInfo(roomID) = (roomName, roomNeededPlayers)
425         userName match {
426           case Some(name) =>
427             roomsUsers(roomID) = Seq(Participant(name, userAddress.get))
428             roomsAddresses(roomID) = Seq(Address(userNotificationAddress.get))
429           case None =>
430             roomsUsers(roomID) = Seq()
431             roomsAddresses(roomID) = Seq()
432         }
433       }
434     }
435 
436     (for (
437       roomUsers <- roomsUsers;
438       roomAddresses <- roomsAddresses if roomUsers._1 == roomAddresses._1;
439       roomInfo <- roomsInfo if roomUsers._1 == roomInfo._1;
440       output = (Room(roomInfo._1, roomInfo._2._1, roomInfo._2._2, roomUsers._2), roomAddresses._2)
441     ) yield output).toSeq
442   }
