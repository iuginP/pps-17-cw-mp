147 class ApiClientActor() extends Actor {
148 
149   /**
150     * Possono essere ricevuti messaggi di tipo [[ApiClientIncomingMessages]] ed inviati quelli di tipo [[ApiClientOutgoingMessages]]
151     *
152     * @return [[Receive]] che gestisce tutti i messaggi corrispondenti alle richieste che è possibile inviare ai servizi online
153     */
154   // Tutti i behaviour sono attivi in contemporanea, la separazione è solo logica per una migliore leggibilità
155   override def receive: Receive = authenticationBehaviour orElse roomManagerBehaviour // orElse ...
156 
157   /*
158    * Behaviour che gestisce tutte le chiamate al servizio di gestione delle stanze.
159    */
160   private val roomApiWrapper = RoomsApiWrapper()
161 
162   import roomApiWrapper._
163 
164   private def roomManagerBehaviour: Receive = {
165     case RoomCreatePrivate(name, nPlayer, token) =>
166       val senderTmp = sender
167       createRoom(name, nPlayer)(token).onComplete({
168         case Success(t) => senderTmp ! RoomCreatePrivateSuccessful(t)
169         case Failure(error) => senderTmp ! RoomCreatePrivateFailure(Option(error.getMessage))
170       })
171     case RoomEnterPrivate(idRoom, address, webAddress, token) =>
172       val senderTmp = sender
173       enterRoom(idRoom, address, webAddress)(token).onComplete({
174         case Success(_) => senderTmp ! RoomEnterPrivateSuccessful
175         case Failure(error) => senderTmp ! RoomEnterPrivateFailure(Option(error.getMessage))
176       })
177     case RoomEnterPublic(nPlayer, address, webAddress, token) =>
178       val senderTmp = sender
179       enterPublicRoom(nPlayer, address, webAddress)(token).onComplete({
180         case Success(_) => senderTmp ! RoomEnterPublicSuccessful
181         case Failure(error) => senderTmp ! RoomEnterPublicFailure(Option(error.getMessage))
182       })
183   }
184 
185   /*
186     * Behavior that handles the calls to the authentication management service.
187     */
188   private val authenticationApiWrapper = AuthenticationApiWrapper()
189 
190   import authenticationApiWrapper._
191 
192   private def authenticationBehaviour: Receive = {
193     case AuthenticationPerformSignIn(username, password) =>
194       val senderTmp = sender
195       login(username, password).onComplete({
196         case Success(token) => senderTmp ! AuthenticationSignInSuccessful(token)
197         case Failure(reason) => senderTmp ! AuthenticationSignInFailure(Option(reason.getMessage))
198       })
199     case AuthenticationPerformSignUp(username, password) =>
200       val senderTmp = sender
201       signUp(username, password).onComplete({
202         case Success(token) => senderTmp ! AuthenticationSignUpSuccessful(token)
203         case Failure(reason) => senderTmp ! AuthenticationSignUpFailure(Option(reason.getMessage))
204       })
205 
206   }
