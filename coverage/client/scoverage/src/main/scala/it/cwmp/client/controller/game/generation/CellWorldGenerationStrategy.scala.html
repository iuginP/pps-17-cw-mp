14 /**
15   * A class implementing a CellWorld generation strategy
16   *
17   * @param worldHeight  the height of the world
18   * @param worldWidth   the width of the world
19   * @param passiveCells the number of passive cells to generate
20   * @author Enrico Siboni
21   */
22 case class CellWorldGenerationStrategy(worldHeight: Int, worldWidth: Int, passiveCells: Int) extends GenerationStrategy[Seq[User], CellWorld] {
23   require(worldHeight > 0, "World height should be greater that 0")
24   require(worldWidth > 0, "World width should be greater that 0")
25   require(passiveCells >= 0, "Passive cells number should be positive")

27   /**
28     * Generates the world starting from a participant list
29     *
30     * @param players the participants to game
31     * @return the generated world starting from participants
32     */
33   override def apply(players: Seq[User]): CellWorld =
34     CellWorld(Instant.now(), generateWorldCells(players, passiveCells), Seq())
35 
36   /**
37     * A method to generate world cells in a way that they don't overlap
38     *
39     * @param players              the participants to game for which to generate cells
40     * @param numberOfPassiveCells the number of passive cells to generate among others
41     * @return generated cells
42     */
43   def generateWorldCells(players: Seq[User], numberOfPassiveCells: Int): Seq[Cell] = {
44     val participantCells = getParticipantCells(players)
45     participantCells ++ randomPassiveCells(numberOfPassiveCells, participantCells.map(_.position))
46   }
47 
48   /**
49     * A method to position participant cells into game world
50     *
51     * @param players the participants to assign a cell into the world
52     * @return the cells assigned to participants
53     */
54   def getParticipantCells(players: Seq[User]): Seq[Cell] = {
55     var assignedPoints: Seq[Point] = Seq()
56     for (player <- players) yield {
57       val validPoint = getValidPoint(assignedPoints)
58       assignedPoints = assignedPoints :+ validPoint
59       Cell(player, validPoint)
60     }
61   }
62 
63   /**
64     * A method to generate a certain number of passive cells to be placed into world
65     *
66     * @param numberOfPassiveCells  the number of passive cells to place
67     * @param alreadyOccupiedPlaces those places already occupied
68     * @return the passive cells placed
69     */
70   def randomPassiveCells(numberOfPassiveCells: Int, alreadyOccupiedPlaces: Seq[Point]): Seq[Cell] = {
71     var assignedPoints: Seq[Point] = alreadyOccupiedPlaces
72     for (_ <- 0 until numberOfPassiveCells) yield {
73       val validPoint = getValidPoint(assignedPoints)
74       assignedPoints = assignedPoints :+ validPoint
75       Cell(Cell.Passive.NO_OWNER, validPoint, GameConstants.PASSIVE_CELL_ENERGY_WHEN_BORN)
76     }
77   }
78 
79   /**
80     * A method to get a valid point for a cell;
81     * that is a point which is not overlapping other cells and distant from world border
82     *
83     * @param alreadySelectedPoints the already selected points
84     * @return the point that is valid according to rules
85     */
86   private def getValidPoint(alreadySelectedPoints: Seq[Point]): Point = {
87     var point = randomPoint(worldWidth, worldHeight)
88     while (!pointValid(point, alreadySelectedPoints)) point = randomPoint(worldWidth, worldHeight)
89     point
90   }
91 
92   /**
93     * Determines if the point in which to place a cell is valid;
94     *
95     * A point is valid when it's distance from borders is respected and also the distance from other cells is respected
96     *
97     * @param toCheck         the point to check
98     * @param alreadySelected the points representing already selected points
99     * @return
100     */
101   private def pointValid(toCheck: Point, alreadySelected: Seq[Point]): Boolean =
102     toCheck.x >= MINIMUM_DISTANCE_FROM_BORDER && toCheck.x <= worldWidth - MINIMUM_DISTANCE_FROM_BORDER &&
103       toCheck.y >= MINIMUM_DISTANCE_FROM_BORDER && toCheck.y <= worldHeight - MINIMUM_DISTANCE_FROM_BORDER &&
104       alreadySelected.forall(point => !GeometricUtils.isWithinCircumference(toCheck, point, MINIMUM_DISTANCE_BETWEEN_CELLS))
105 }

114   /**
115     * Returns a random point with coordinates between 0 and max values (inclusive)
116     *
117     * @param maxX the max X of any generated point
118     * @param maxY the max Y of any generated point
119     * @return the generated point
120     */
121   private def randomPoint(maxX: Int, maxY: Int): Point = Point(Random.nextInt(maxX + 1), Random.nextInt(maxY + 1))
