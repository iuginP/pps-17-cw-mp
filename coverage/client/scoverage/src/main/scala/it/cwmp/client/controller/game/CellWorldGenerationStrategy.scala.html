13 /**
14   * A class implementing a CellWorld generation strategy
15   *
16   * @param worldHeight  the height of the world
17   * @param worldWidth   the width of the world
18   * @param passiveCells the number of passive cells to generate

21 case class CellWorldGenerationStrategy(worldHeight: Int, worldWidth: Int, passiveCells: Int) extends GenerationStrategy[Seq[User], CellWorld] {
22   require(worldHeight > 0, "World height should be greater that 0")
23   require(worldWidth > 0, "World width should be greater that 0")
24   require(passiveCells >= 0, "Passive cells number should be positive")

26   /**
27     * Generates the world starting from a participant list
28     *
29     * @param players the participants to game
30     * @return the generated world starting from participants
31     */
32   override def apply(players: Seq[User]): CellWorld =
33     CellWorld(Instant.now(), generateWorldCells(players, passiveCells), Seq())
34 
35   /**
36     * A method to generate world cells in a way that they don't overlap
37     *
38     * @param players              the participants to game for which to generate cells
39     * @param numberOfPassiveCells the number of passive cells to generate among others
40     * @return generated cells
41     */
42   def generateWorldCells(players: Seq[User], numberOfPassiveCells: Int): Seq[Cell] = {
43     val participantCells = getParticipantCells(players)
44     participantCells ++ randomPassiveCells(numberOfPassiveCells, participantCells.map(_.position))
45   }
46 
47   /**
48     * A method to position participant cells into game world
49     *
50     * @param players the participants to assign a cell into the world
51     * @return the cells assigned to participants
52     */
53   def getParticipantCells(players: Seq[User]): Seq[Cell] = {
54     var assignedPoints: Seq[Point] = Seq()
55     for (player <- players) yield {
56       val validPoint = getValidPoint(assignedPoints)
57       assignedPoints = assignedPoints :+ validPoint
58       Cell(player, validPoint)
59     }
60   }
61 
62   /**
63     * A method to generate a certain number of passive cells to be placed into world
64     *
65     * @param numberOfPassiveCells  the number of passive cells to place
66     * @param alreadyOccupiedPlaces those places already occupied
67     * @return the passive cells placed
68     */
69   def randomPassiveCells(numberOfPassiveCells: Int, alreadyOccupiedPlaces: Seq[Point]): Seq[Cell] = {
70     var assignedPoints: Seq[Point] = alreadyOccupiedPlaces
71     for (_ <- 0 until numberOfPassiveCells) yield {
72       val validPoint = getValidPoint(assignedPoints)
73       assignedPoints = assignedPoints :+ validPoint
74       Cell(Cell.Passive.NO_OWNER, validPoint, GameConstants.PASSIVE_CELL_ENERGY_WHEN_BORN)
75     }
76   }
77 
78   /**
79     * A method to get a valid point for a cell;
80     * that is a point which is not overlapping other cells and distant from world border
81     *
82     * @param alreadySelectedPoints the already selected points
83     * @return the point that is valid according to rules
84     */
85   private def getValidPoint(alreadySelectedPoints: Seq[Point]): Point = {
86     var point = randomPoint(worldWidth, worldHeight)
87     while (!pointValid(point, alreadySelectedPoints)) point = randomPoint(worldWidth, worldHeight)
88     point
89   }
90 
91   /**
92     * Determines if the point in which to place a cell is valid;
93     *
94     * A point is valid when it's distance from borders is respected and also the distance from other cells is respected
95     *
96     * @param toCheck         the point to check
97     * @param alreadySelected the points representing already selected points
98     * @return
99     */
100   private def pointValid(toCheck: Point, alreadySelected: Seq[Point]): Boolean =
101     toCheck.x >= MINIMUM_DISTANCE_FROM_BORDER && toCheck.x <= worldWidth - MINIMUM_DISTANCE_FROM_BORDER &&
102       toCheck.y >= MINIMUM_DISTANCE_FROM_BORDER && toCheck.y <= worldHeight - MINIMUM_DISTANCE_FROM_BORDER &&
103       alreadySelected.forall(point => !GeometricUtils.isWithinCircumference(toCheck, point, MINIMUM_DISTANCE_BETWEEN_CELLS))
104 }

113   /**
114     * Returns a random point with coordinates between 0 and max values (inclusive)
115     *
116     * @param maxX the max X of any generated point
117     * @param maxY the max Y of any generated point
118     * @return the generated point
119     */
120   private def randomPoint(maxX: Int, maxY: Int): Point = Point(Random.nextInt(maxX + 1), Random.nextInt(maxY + 1))
