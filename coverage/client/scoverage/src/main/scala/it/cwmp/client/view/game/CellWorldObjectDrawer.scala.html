78   /**
79     * A method to draw elapsed time on GUI
80     *
81     * @param actualWorldInstant the actual World Instant
82     * @param graphicsContext    the graphic context on which to draw
83     * @return the text to draw
84     */
85   def drawInstant(actualWorldInstant: Instant)
86                  (implicit graphicsContext: GraphicsContext): Text = {
87     val elapsedTimeFromBeginning: Duration = firstWorldInstantOption match {
88       case Some(firstWorldInstant) => Duration.between(firstWorldInstant, actualWorldInstant)
89       case None =>
90         firstWorldInstantOption = Some(actualWorldInstant)
91         Duration.ofSeconds(0)
92     }
93     val minutes = elapsedTimeFromBeginning.getSeconds / 60
94     val seconds = elapsedTimeFromBeginning.getSeconds % 60
95     val instantText = new Text(GameViewConstants.GAME_TIME_TEXT_FORMAT.format(minutes, seconds))
96     instantText.setFont(GameViewConstants.GAME_TIME_TEXT_FONT)
97     instantText.setFill(GameViewConstants.GAME_TIME_TEXT_COLOR)
98     instantText.setX((graphicsContext.getCanvas.getWidth / 2) - (instantText.getLayoutBounds.getWidth / 2))
99     instantText.setY(instantText.getLayoutBounds.getHeight)
100     instantText
101   }
