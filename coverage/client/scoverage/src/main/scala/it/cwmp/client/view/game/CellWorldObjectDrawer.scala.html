23   /**

27     * @return the drawn region
28     */
29   def drawCell(cell: CellView): Region = {
30     val svg = new SVGPath
31     svg.setContent(CellView.CELL_VIEW_DEFAULT_SVG_SHAPE)
32     val svgShape = new Region
33     svgShape.setShape(svg)
34     svgShape.setBorder(cell.border)
35     svgShape.setPrefSize(cell.radius * 2, cell.radius * 2)
36     svgShape.setStyle(getCellViewBackgroundStyle(cell.color))

38     svgShape.setLayoutY(cell.center.y - cell.radius)
39     svgShape
40   }

42   /**

44     *

46     * @return the text to add to scene

49     val energyText = new Text(cell.energy.toInt.toString)
50     energyText.setFont(CellView.ENERGY_DEFAULT_FONT)
51     energyText.setFill(cell.energyTextColor)
52     energyText.setX(cell.center.x - (energyText.getLayoutBounds.getWidth / 2))
53     energyText.setY(cell.center.y + (energyText.getLayoutBounds.getHeight / 2))

55   }

77   /**
78     * A method to draw elapsed time on GUI
79     *
80     * @param actualWorldInstant the actual World Instant
81     * @param viewWidth          the view Width
82     * @return the text to draw
83     */
84   def drawInstant(actualWorldInstant: Instant, viewWidth: Double): Text = {
85     val elapsedTimeFromBeginning: Duration = firstWorldInstantOption match {
86       case Some(firstWorldInstant) => Duration.between(firstWorldInstant, actualWorldInstant)
87       case None =>
88         firstWorldInstantOption = Some(actualWorldInstant)
89         Duration.ofSeconds(0)
90     }
91     val minutes = elapsedTimeFromBeginning.getSeconds / 60
92     val seconds = elapsedTimeFromBeginning.getSeconds % 60
93     val instantText = new Text(GameViewConstants.GAME_TIME_TEXT_FORMAT.format(minutes, seconds))
94     instantText.setFont(GameViewConstants.GAME_TIME_TEXT_FONT)
95     instantText.setFill(GameViewConstants.GAME_TIME_TEXT_COLOR)
96     instantText.setX((viewWidth / 2) - (instantText.getLayoutBounds.getWidth / 2))
97     instantText.setY(instantText.getLayoutBounds.getHeight)
98     instantText
99   }
