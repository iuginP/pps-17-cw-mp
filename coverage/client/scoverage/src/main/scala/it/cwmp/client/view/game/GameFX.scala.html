93     /**
94       * A method to initialize event handlers for GUI user actions
95       *
96       * @param viewGroup the viewGroup on which to listen for events
97       */
98     def initializeEventHandlers(viewGroup: Group, viewManagerActor: ActorRef): Unit = {
99 
100       // user pressed mouse
101       viewGroup.addEventHandler(MouseEvent.MOUSE_PRESSED, (event: MouseEvent) => startDragPoint = event)
102 
103       // start of user dragging
104       viewGroup.addEventHandler(MouseEvent.DRAG_DETECTED, (_: MouseEvent) => mouseIsDragging = true)
105 
106       // stop of user dragging
107       viewGroup.addEventHandler(MouseEvent.MOUSE_RELEASED, (event: MouseEvent) =>
108         if (mouseIsDragging) {
109           val stopDragPoint: Point = event
110           sendAddAttackEvent(startDragPoint, stopDragPoint, viewManagerActor)
111         })
112 
113       // user click event
114       viewGroup.addEventHandler(MouseEvent.MOUSE_CLICKED, (event: MouseEvent) =>
115         // if the user was dragging this event is launched after MOUSE_RELEASED
116         if (mouseIsDragging) {
117           mouseIsDragging = false // reset user dragging state
118         } else {
119           sendRemoveAttackEvent(event, viewManagerActor)
120         }
121       )
122     }
123 
124     /**
125       * A method to send the AddAttack message to provided actor
126       *
127       * @param start the start point of the attack
128       * @param stop  the stop point of the attack
129       * @param actor the actor responsible of this management
130       */
131     private def sendAddAttackEvent(start: Point, stop: Point, actor: ActorRef): Unit =
132       actor ! GameViewActor.AddAttack(start, stop)

134     /**

138       * @param actor         the actor responsible of this management
139       */
140     private def sendRemoveAttackEvent(onAttackPoint: Point, actor: ActorRef): Unit =
141       actor ! GameViewActor.RemoveAttack(onAttackPoint)
142

148       */
149     implicit def eventToPoint(event: MouseEvent): Point =
150       Point(event.getX.toInt, event.getY.toInt)
151   }
