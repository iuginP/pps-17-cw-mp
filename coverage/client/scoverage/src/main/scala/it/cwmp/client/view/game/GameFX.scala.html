96     /**
97       * A method to initialize event handlers for GUI user actions
98       *
99       * @param viewGroup the viewGroup on which to listen for events
100       */
101     def initializeEventHandlers(viewGroup: Group, viewManagerActor: ActorRef): Unit = {
102 
103       // user pressed mouse
104       viewGroup.addEventHandler(MouseEvent.MOUSE_PRESSED, (event: MouseEvent) => startDragPoint = event)
105 
106       // start of user dragging
107       viewGroup.addEventHandler(MouseEvent.DRAG_DETECTED, (_: MouseEvent) => mouseIsDragging = true)
108 
109       // stop of user dragging
110       viewGroup.addEventHandler(MouseEvent.MOUSE_RELEASED, (event: MouseEvent) =>
111         if (mouseIsDragging) {
112           val stopDragPoint: Point = event
113           sendAddAttackEvent(startDragPoint, stopDragPoint, viewManagerActor)
114         })
115 
116       // user click event
117       viewGroup.addEventHandler(MouseEvent.MOUSE_CLICKED, (event: MouseEvent) =>
118         // if the user was dragging this event is launched after MOUSE_RELEASED
119         if (mouseIsDragging) {
120           mouseIsDragging = false // reset user dragging state
121         } else {
122           sendRemoveAttackEvent(event, viewManagerActor)
123         }
124       )
125     }
126 
127     /**
128       * A method to send the AddAttack message to provided actor
129       *
130       * @param start the start point of the attack
131       * @param stop  the stop point of the attack
132       * @param actor the actor responsible of this management
133       */
134     private def sendAddAttackEvent(start: Point, stop: Point, actor: ActorRef): Unit =
135       actor ! GameViewActor.AddAttack(start, stop)

137     /**

139       *
140       * @param onAttackPoint the point on the Attack View to remove
141       * @param actor         the actor responsible of this management
142       */
143     private def sendRemoveAttackEvent(onAttackPoint: Point, actor: ActorRef): Unit =
144       actor ! GameViewActor.RemoveAttack(onAttackPoint)
145

151       */
152     implicit def eventToPoint(event: MouseEvent): Point =
153       Point(event.getX.toInt, event.getY.toInt)
154   }
