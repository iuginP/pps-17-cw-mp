68   /**
69     * An object that wraps logic behind user events un GUI
70     *
71     * @author Enrico Siboni
72     */
73   private object UserEventHandler {
74 
75     private var mouseIsDragging = false
76     private var startDragPoint: Point = _
77 
78     /**
79       * A method to initialize event handlers for GUI user actions
80       *
81       * @param viewGroup the viewGroup on which to listen for events
82       */
83     def initializeEventHandlers(viewGroup: Group, viewManagerActor: ActorRef): Unit = {
84 
85       // user pressed mouse
86       viewGroup.addEventHandler(MouseEvent.MOUSE_PRESSED, (event: MouseEvent) => startDragPoint = event)
87 
88       // start of user dragging
89       viewGroup.addEventHandler(MouseEvent.DRAG_DETECTED, (_: MouseEvent) => mouseIsDragging = true)
90 
91       // stop of user dragging
92       viewGroup.addEventHandler(MouseEvent.MOUSE_RELEASED, (event: MouseEvent) =>
93         if (mouseIsDragging) {
94           val stopDragPoint: Point = event
95           sendAddAttackEvent(startDragPoint, stopDragPoint, viewManagerActor)
96         })
97 
98       // user click event
99       viewGroup.addEventHandler(MouseEvent.MOUSE_CLICKED, (event: MouseEvent) =>
100         // if the user was dragging this event is launched after MOUSE_RELEASED
101         if (mouseIsDragging) {
102           mouseIsDragging = false // reset user dragging state
103         } else {
104           sendRemoveAttackEvent(event, viewManagerActor)
105         }
106       )
107     }
108 
109     /**
110       * A method to send the AddAttack message to provided actor
111       *
112       * @param start the start point of the attack
113       * @param stop  the stop point of the attack
114       * @param actor the actor responsible of this management
115       */
116     private def sendAddAttackEvent(start: Point, stop: Point, actor: ActorRef): Unit =
117       actor ! GameViewActor.AddAttack(start, stop)
118 
119     /**
120       * A method to send the RemoveAttack messsage to provided actor
121       *
122       * @param onAttackPoint the point on the Attack View to remove
123       * @param actor         the actor responsible of this management
124       */
125     private def sendRemoveAttackEvent(onAttackPoint: Point, actor: ActorRef): Unit =
126       actor ! GameViewActor.RemoveAttack(onAttackPoint)
127 
128     /**

132       * @return the Point where event was generated
133       */
134     implicit def eventToPoint(event: MouseEvent): Point =
135       Point(event.getX.toInt, event.getY.toInt)
136   }
