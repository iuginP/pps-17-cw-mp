141   /**
142     * @return the behaviour that manages rooms commands from GUI
143     */
144   private def roomsGUIBehaviour: Receive = {
145     case GUICreate(roomName, playersNumber) =>
146       log.info(s"Creating the room $roomName")
147       apiClientActor ! ServiceCreate(roomName, playersNumber, jwtToken)
148     case GUIEnterPrivate(roomID) =>
149       log.info(s"Entering the private room $roomID")
150       openOneTimeServerAndGetAddress()
151         .map(url => apiClientActor ! ServiceEnterPrivate(roomID, Address(playerAddress), url, jwtToken))
152     case GUIEnterPublic(playersNumber) =>
153       log.info(s"Entering the public room with $playersNumber players")
154       openOneTimeServerAndGetAddress()
155         .map(url => apiClientActor ! ServiceEnterPublic(playersNumber, Address(playerAddress), url, jwtToken))
156     case GUIExitPrivate(roomID) =>
157       log.info(s"Exiting room $roomID")
158       apiClientActor ! ServiceExitPrivate(roomID, jwtToken)
159     case GUIExitPublic(playersNumber) =>
160       log.info(s"Exiting public room with $playersNumber players")
161       apiClientActor ! ServiceExitPublic(playersNumber, jwtToken)
162     case GUILogOut =>
163       log.info("Logging-out")
164       onLogOut()
165   }

167   /**
168     * @return the behaviour that manages room API responses

181     case ExitPublicSuccess => onRoomExitingSuccess()

183     case ExitPublicFailure(errorMessage) => onRoomExitingFailure(errorMessage)

245   /**
246     * @return the Future containing the address of one-time server that will receive the participants
247     */
248   private def openOneTimeServerAndGetAddress(): Future[Address] = {
249     val onListReceived: List[Participant] => Unit = participants => {
250       log.info(s"Participants list received!")
251       onSuccessFindingOpponents(participants)
252     }
253 
254     log.debug(s"Starting the local one-time receiver server...")
255     listenForParticipantListFuture(onListReceived)
256       .andThen {
257         case Success(address) =>
258           log.debug(s"Server started and listening at the address: $address")
259         case Failure(error) =>
260           log.error(s"Problem starting the server: ${error.getMessage}")
261           roomViewActor ! Error(RECEIVING_PARTICIPANT_LIST_ERROR_TITLE, error.getMessage)
262       }
263   }
264 
265   /**
266     * Action to execute when found opponents

270   private def onSuccessFindingOpponents(participants: List[Participant]): Unit = {
271     roomViewActor ! FoundOpponents

273     context.become(inGameBehaviour)
274     roomViewActor ! Hide
275     playerActor ! PrepareForGame(participants,
276       CellWorldGenerationStrategy(GameViewActor.VIEW_SIZE, GameViewActor.VIEW_SIZE, GameConstants.PASSIVE_CELLS_NUMBER))
277   }
