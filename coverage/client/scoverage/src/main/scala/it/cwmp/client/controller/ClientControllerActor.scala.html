205   /**
206     * @return the Future containing the address of one-time server that will receive the participants
207     */
208   private def openOneTimeServerAndGetAddress(): Future[Address] = {
209     val onListReceived: List[Participant] => Unit = participants => {
210       log.info(s"Participants list received!")
211       onSuccessFindingOpponents(participants)
212     }
213 
214     log.debug(s"Starting the local one-time receiver server...")
215     listenForParticipantListFuture(onListReceived)
216       .andThen({
217         case Success(address) =>
218           log.debug(s"Server started and listening at the address: $address")
219         case Failure(error) =>
220           log.error(s"Problem starting the server: ${error.getMessage}")
221           roomViewActor ! Error(RECEIVING_PARTICIPANT_LIST_ERROR_TITLE, error.getMessage)
222       })
223   }
