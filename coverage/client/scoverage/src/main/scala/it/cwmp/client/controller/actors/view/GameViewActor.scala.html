57     case ShowGUIWithName(name) =>

68   /**
69     * The behaviour of receiving world modifications from external source
70     */
71   private def newWorldBehaviour: Receive = {
72     case NewWorld(world) =>
73       tempWorld = world
74       restartSynchronizationSchedule(WAIT_TIME_BEFORE_AUTOMATIC_SYNCHRONIZATION)
75       if (updatingSchedule == null) {
76         updatingSchedule = context.system.scheduler
77           .schedule(TIME_BETWEEN_FRAMES, TIME_BETWEEN_FRAMES, self, UpdateGUI)(context.dispatcher)
78       }
79       gameEnded(world) match {
80         case Some(winnerName) if winnerName == playerName => self ! AlertMessages.Info(s"$playerName $YOU_WON_TITLE", YOU_WON)
81         case Some(_) => self ! AlertMessages.Info(s"$playerName $YOU_LOST_TITLE", YOU_LOST)
82         case None =>
83       }
84 
85     case UpdateGUI =>
86       Future {
87         tempWorld = GameEngine(tempWorld, java.time.Duration.ofMillis(TIME_BETWEEN_FRAMES.toMillis))
88         tempWorld
89       } andThen {
90         case Success(cellWorld) => runOnUIThread { () => gameFX.updateWorld(cellWorld) }
91         case Failure(ex) =>
92           updatingSchedule.cancel()
93           log.error("Error calculating next CellWorld", ex)
94       }
95   }
96 
97   /**
98     * The behaviour of listening for user events on GUI
99     */
100   private def guiWorldModificationsBehaviour: Receive = {
101     case AddAttack(from, to) =>
102       log.info(s"AddAttack from:$from to:$to")
103       val worldCharacters = tempWorld.characters
104       val fromCell = findCellNearTo(from, worldCharacters)
105       val toCell = findCellNearTo(to, worldCharacters)
106       (fromCell, toCell) match {
107         case (Some(attacker), Some(attacked)) if canAddAttack(attacker, attacked, tempWorld.attacks) =>
108           log.debug(s"Adding attack from $attacker to $attacked ...")
109           parentActor ! UpdateState(tempWorld ++ Tentacle(attacker, attacked, tempWorld.instant))
110           restartSynchronizationSchedule(WAIT_TIME_BEFORE_AUTOMATIC_SYNCHRONIZATION)
111         case tmp@_ => log.debug(s"No cells detected or auto-attack or not $playerName cell $tmp")
112       }
113 
114     case RemoveAttack(pointOnAttackView) =>
115       log.info(s"RemoveAttack pointOnView:$pointOnAttackView")
116       val attacks = findTentaclesNearTo(pointOnAttackView, tempWorld.attacks)
117       attacks.find(_.from.owner.username == playerName) match {
118         case Some(tentacle) =>
119           log.debug(s"Removing this attack: $tentacle ...")
120           parentActor ! UpdateState(tempWorld -- tentacle)
121           restartSynchronizationSchedule(WAIT_TIME_BEFORE_AUTOMATIC_SYNCHRONIZATION)
122         case tmp@_ => log.debug(s"No attack detected or not $playerName attack $tmp")
123       }
124 
125     case SynchronizeWorld =>
126       log.info("Time to synchronize other clients because of no user actions")
127       parentActor ! UpdateState(tempWorld)
128   }

130   /**
131     * Utility method to know if can add attack
132     *
133     * @param attacker       the attacker cell
134     * @param attacked       the attacked cell
135     * @param currentAttacks the current game attacks
136     * @return true if attack can be added, false otherwise
137     */
138   private def canAddAttack(attacker: Cell, attacked: Cell, currentAttacks: Seq[Tentacle]): Boolean = {
139     !Cell.ownerAndPositionMatch(attacker, attacked) && // no auto-attack
140       attacker.owner.username == playerName && // control only your cells
141       !currentAttacks.exists(tentacle => // no already present attacks
142         Cell.ownerAndPositionMatch(tentacle.from, attacker) &&
143           Cell.ownerAndPositionMatch(tentacle.to, attacked))
144   }
145 
146   /**
147     * Utility method to restart the world synchronization schedule
148     */
149   private def restartSynchronizationSchedule(waitTime: FiniteDuration): Unit = {
150     if (synchronizationSchedule != null) synchronizationSchedule.cancel()
151 
152     synchronizationSchedule = context.system.scheduler
153       .schedule(waitTime, waitTime, self, SynchronizeWorld)(context.dispatcher)
154   }

156   /**
157     * Checks if the game has ended
158     *
159     * @param cellWorld the cellWorld to check
160     * @return optionally the winner name
161     */
162   private def gameEnded(cellWorld: CellWorld): Option[String] =
163     if (cellWorld.characters.forall(_.owner == cellWorld.characters.head.owner)) {
164       cellWorld.characters.head.owner.username
165     } else None
166 }

233   private def findCellNearTo(clickedPoint: Point, cells: Seq[Cell]): Option[Cell] =
234     cells.find(cell => GeometricUtils.isWithinCircumference(clickedPoint, cell.position, CellView.sizingStrategy(cell)))
