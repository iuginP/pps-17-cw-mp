23     * Listens for a list of participants
24     *
25     * @param onListReceived the action to execute on list received
26     * @return the Future containing the address on which to contact this listener
27     */
28   def listenForParticipantListFuture(onListReceived: List[Participant] => Unit): Future[Address] = {
29     val token = Utils.randomString(ADDRESS_TOKEN_LENGTH)
30     val verticle = RoomReceiverServiceVerticle(token, participants => onListReceived(participants))
31     vertx.deployVerticleFuture(verticle)
32       .andThen { case Success(id) => deploymentID = id }
33       .map(_ => Address(s"http://${InetAddress.getLocalHost.getHostAddress}:${verticle.port}"
34         + createParticipantReceiverUrl(token)))
35   }
36 
37   /**
38     * A method that stops the server that would have received participants to entered room
39     */
40   def stopListeningForParticipants(): Unit = {
41     log.info("Stopping one-time participants receiver server")
42     deploymentID.foreach(vertx.undeploy)
43   }
44 }
