30     val token = Utils.randomString(ADDRESS_TOKEN_LENGTH)
31     val verticle = RoomReceiverServiceVerticle(token, participants => onListReceived(participants))
32     vertx.deployVerticleFuture(verticle)

34       .map(_ => Address(s"http://$hostname:${verticle.port}"
35         + createParticipantReceiverUrl(token)))
36   }

40     */
41   def stopListeningForParticipants(): Unit = {
