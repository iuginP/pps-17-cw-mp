11 /**

44   protected def consistencyPolicy: Replicator.WriteConsistency = WriteMajority(1.seconds)

52       // Called when notified of the distributed data change
53       case msg@Changed(`changedDistributedKey`) =>

55         onDistributedStateUpdate(msg.get[ReplicatedDataType](changedDistributedKey))
56     }
57   }

74   protected def updateDistributedStateTo(state: State)

76   /**
77     * A function to modify distributed state to the new one

79     * @param oldDistributedState the old distributed state to modify
80     * @param newState            the new state to inject

83   protected def distributedModify(oldDistributedState: ReplicatedDataType, newState: State): ReplicatedDataType
