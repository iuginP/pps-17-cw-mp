97     runOnUIThread(() => {

100       alert setHeaderText (if (dialogContent.isDefined) s"$headerText\n\n$message" else headerText)
101       dialogContent foreach (alert.getDialogPane.setContent(_))
102       alert.showAndWait

113     /**
114       * Creates and shows a non blocking loading alert
115       *
116       * @param title   the title of the loading alert
117       * @param message the message to show in the loading alert
118       */
119     def createLoadingAlert(title: String, message: String,
120                            canBeClosed: Boolean,
121                            onPrematureCloseAction: () => Unit = () => ()): Alert = {
122       loadingAlert = new Alert(AlertType.NONE)
123       loadingAlert setTitle title
124       loadingAlert setHeaderText message
125       val pane = new BorderPane()
126       pane.setCenter(new ProgressBar())
127       loadingAlert.getDialogPane.setContent(pane)
128 
129       if (canBeClosed) {
130         loadingAlert.getButtonTypes.add(ButtonType.CANCEL)
131         loadingAlert.setOnCloseRequest(_ => onPrematureCloseAction())
132       }
133 
134       loadingAlert
135     }
136 
137     /**

140     def closeLoadingAlert(): Unit =
141       runOnUIThread(() => {
142         //noinspection ScalaStyle
143         loadingAlert.setOnCloseRequest(null)
144         loadingAlert.setResult(ButtonType.OK) // setting a result of right type, seems the only way to have dialog to close
145       })
146   }
