78     * Color based on the username hash value
79     */
80   val coloringStrategy: ColoringStrategy[Cell, Color] = (cell: Cell) => {
81     if (Cell.isPassiveCell(cell)) {
82       PASSIVE_CELL_COLOR
83     } else {
84       implicit def colorFromRGB(userColor: Rgb): Color =
85         new Color(userColor.red / RGB_RANGE, userColor.green / RGB_RANGE, userColor.blue / RGB_RANGE, CELL_VIEW_COLOR_OPACITY)
86 
87       val colorHash = new ColorHash()
88       val username = cell.owner.username
89       val userColor: Color = colorHash.rgb(username)
90       // if userName hash gives a color used in GUI take another color
91       if (Seq(GAME_DEFAULT_FONT_COLOR, CELL_DYING_FONT_COLOR, GAME_TIME_TEXT_COLOR, PASSIVE_CELL_COLOR).contains(userColor)) {
92         colorHash.rgb(username.substring(username.length / 2))
93       } else {
94         userColor
95       }
96     }

126   /**
127     * A method to calculate the radius of the cell between maximum and minimum provided
128     *
129     * @param cell                   the cell to draw
130     * @param minimumRadiusAndEnergy the lower bound values
131     * @param maximumRadiusAndEnergy the upper bound values
132     * @return the sized cell radius
133     */
134   private def radiusBetweenMinimumAndMaximum(cell: Cell,
135                                              minimumRadiusAndEnergy: (Double, Double),
136                                              maximumRadiusAndEnergy: (Double, Double)): Double = {
137     val energyDeltaFromMinimum = cell.energy - minimumRadiusAndEnergy._2
138     val minimumPoint = Point(minimumRadiusAndEnergy._1.toInt, minimumRadiusAndEnergy._2.toInt)
139     val maximumPoint = Point(maximumRadiusAndEnergy._1.toInt, maximumRadiusAndEnergy._2.toInt)
140 
141     // I use geometric utils to calculate a pair (a point) between minimum and maximum pairs, related to energy delta
142     // I'm interested in first component of the delta because it represents the radius
143     val radiusDelta = GeometricUtils.deltaXYFromFirstPoint(minimumPoint, maximumPoint, energyDeltaFromMinimum)._1
144     minimumRadiusAndEnergy._1 + radiusDelta
145   }
