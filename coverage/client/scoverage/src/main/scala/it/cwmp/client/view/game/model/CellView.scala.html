104     */
105   val sizingStrategy: SizingStrategy[Cell, Double] = {
106     case cell: Cell if cell.energy <= CELL_MINIMUM_RADIUS_FOR_ENERGY._2 => CELL_MINIMUM_RADIUS_FOR_ENERGY._1
107     case cell: Cell if cell.energy >= CELL_MAX_RADIUS_FOR_ENERGY._2 => CELL_MAX_RADIUS_FOR_ENERGY._1
108     case cell: Cell => radiusBetweenMinimumAndMaximum(cell, CELL_MINIMUM_RADIUS_FOR_ENERGY, CELL_MAX_RADIUS_FOR_ENERGY)
109   }

111   /**
112     * A method to calculate the radius of the cell between maximum and minimum provided
113     *
114     * @param cell                   the cell to draw
115     * @param minimumRadiusAndEnergy the lower bound values
116     * @param maximumRadiusAndEnergy the upper bound values
117     * @return the sized cell radius
118     */
119   private def radiusBetweenMinimumAndMaximum(cell: Cell,
120                                              minimumRadiusAndEnergy: (Double, Double),
121                                              maximumRadiusAndEnergy: (Double, Double)): Double = {
122     val energyDeltaFromMinimum = cell.energy - minimumRadiusAndEnergy._2
123     val minimumPoint = Point(minimumRadiusAndEnergy._1.toInt, minimumRadiusAndEnergy._2.toInt)
124     val maximumPoint = Point(maximumRadiusAndEnergy._1.toInt, maximumRadiusAndEnergy._2.toInt)
125 
126     // I use geometric utils to calculate a pair (a point) between minimum and maximum pairs, related to energy delta
127     // I'm interested in first component of the delta because it represents the radius
128     val radiusDelta = GeometricUtils.deltaXYFromFirstPoint(minimumPoint, maximumPoint, energyDeltaFromMinimum)._1
129     minimumRadiusAndEnergy._1 + radiusDelta
130   }
