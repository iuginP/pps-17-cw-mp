101   /**
102     * A method to calculate the radius of the cell between maximum and minimum provided
103     *
104     * @param cell                   the cell to draw
105     * @param minimumRadiusAndEnergy the lower bound values
106     * @param maximumRadiusAndEnergy the upper bound values
107     * @return the sized cell radius
108     */
109   private def radiusBetweenMinimumAndMaximum(cell: Cell,
110                                              minimumRadiusAndEnergy: (Double, Double),
111                                              maximumRadiusAndEnergy: (Double, Double)): Double = {
112     val energyDeltaFromMinimum = cell.energy - minimumRadiusAndEnergy._2
113     val minimumPoint = Point(minimumRadiusAndEnergy._1.toInt, minimumRadiusAndEnergy._2.toInt)
114     val maximumPoint = Point(maximumRadiusAndEnergy._1.toInt, maximumRadiusAndEnergy._2.toInt)
115 
116     // I use geometric utils to calculate a pair (a point) between minimum and maximum pairs, related to energy delta
117     // I'm interested in first component of the delta because it represents the radius
118     val radiusDelta = GeometricUtils.deltaXYFromFirstPoint(minimumPoint, maximumPoint, energyDeltaFromMinimum)._1
119     minimumRadiusAndEnergy._1 + radiusDelta
120   }
