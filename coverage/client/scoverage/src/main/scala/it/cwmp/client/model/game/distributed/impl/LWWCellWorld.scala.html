24 case class LWWCellWorld(onWorldUpdate: CellWorld => Unit)(implicit replicatorActor: ActorRef, cluster: Cluster)
25   extends AkkaDistributedState[CellWorld](onWorldUpdate) {
26 
27   override type ReplicatedDataType = LWWRegister[CellWorld]
28 
29   override protected val distributedKey: LWWRegisterKey[CellWorld] = LWWRegisterKey(DISTRIBUTED_KEY_NAME)
30 
31   override protected def activeBehaviour: Receive = {
32     case UpdateState(state: CellWorld) =>
33       log.debug("Updating distributed state")
34       updateDistributedStateTo(state)
35   }
36 
37   override protected def updateDistributedStateTo(state: CellWorld): Unit =
38     replicatorActor ! Update(distributedKey, LWWRegister(state), consistencyPolicy)(distributedModify(_, state))
39 
40   override protected def distributedModify(oldDistributedState: LWWRegister[CellWorld], newState: CellWorld): LWWRegister[CellWorld] =
41     oldDistributedState.withValue(newState)
42 
43   override protected def parseFromDistributed(distributedData: LWWRegister[CellWorld]): CellWorld = distributedData.value
44 }
