12   /**
13     * Calculates the distance between two points
14     *
15     * @param point1 the first point
16     * @param point2 the second point
17     * @return the distance
18     */
19   def distance(point1: Point, point2: Point): Double = {
20     Math.sqrt {
21       (point1.x.toDouble - point2.x).squared +
22         (point1.y.toDouble - point2.y).squared
23     }
24   }

26   /**
27     * Calculates the angular coefficient of a straight line passing through two points
28     *
29     * m in this formula -> y = mx + q
30     *
31     * @param point1 the first point
32     * @param point2 the second point
33     * @return the angular coefficient or [[Double.PositiveInfinity]] if the straight line is vertical
34     */
35   def angularCoefficient(point1: Point, point2: Point): Double = {
36     if (point1.x == point2.x) Double.PositiveInfinity
37     else (point1.y.toDouble - point2.y) / (point1.x.toDouble - point2.x)
38   }

40   /**
41     * Calculates the ordinate at the origin of a straight line passing through two points
42     *
43     * q in this formula -> y = mx + q
44     *
45     * @param point1 the first point
46     * @param point2 the second point
47     * @return the ordinate at the origin or [[Double.NaN]] if the straight line is vertical
48     */
49   def ordinateAtOrigin(point1: Point, point2: Point): Double = {
50     if (point1.x == point2.x) Double.NaN
51     else (point1.x.toDouble * point2.y - point2.x.toDouble * point1.y) / (point1.x.toDouble - point2.x)
52   }

54   /**
55     * Calculates the delta X and Y starting from point1 of a point that will be on a straight line passing through two point at given distance from first point
56     *
57     * @param point1   the point from which to refer for the distance
58     * @param point2   the other point from which the straight line should pass
59     * @param distance the distance from point1 on the straight line towards point2
60     * @return a couple of numbers that represents the delta X and Y from point1
61     *         where will be placed a point at given distance on the straight line
62     */
63   def deltaXYFromFirstPoint(point1: Point,
64                             point2: Point,
65                             distance: Double): (Double, Double) = {
66     require(distance > 0, "Distance should be greater than 0")
67 
68     val deltaY =
69     // point on same horizontal line has no delta Y
70       if (point1.y == point2.y) 0
71       // point on same vertical line has delta Y that equals distance to travel;
72       // signum function makes delta positive if first point below second, negative otherwise
73       else if (point1.x == point2.x) distance * Math.signum(point2.y.toLong - point1.y)
74       // this is the formula to get deltaY from a distance, the signum function has same significance as in above comment
75       else distance / Math.sqrt(1 / angularCoefficient(point1, point2).squared + 1) * Math.signum(point2.y.toLong - point1.y)
76 
77     val deltaX =
78     // point on same vertical line has no delta X
79       if (point1.x == point2.x) 0
80       // point on same horizontal line has delta X that equals distance to travel;
81       // signum function makes delta positive if first point is on the left of the second, negative otherwise
82       else if (point1.y == point2.y) distance * Math.signum(point2.x.toLong - point1.x)
83       // this is the formula to get deltaX from a distance, the signum function has same significance as in above comment
84       else distance / Math.sqrt(angularCoefficient(point1, point2).squared + 1) * Math.signum(point2.x.toLong - point1.x)
85 
86     (deltaX, deltaY)
87   }

89   /**
90     * A method returning the distance of a point from a straight line passing through two points
91     *
92     * @param myPoint the point from which to calculate the distance
93     * @param point1  the first point from which the straight line is passing through
94     * @param point2  the second point from which the straight line is passing through
95     * @return the distance of first point from this straight line
96     */
97   def pointDistanceFromStraightLine(myPoint: Point,
98                                     point1: Point,
99                                     point2: Point): Double = {
100     val angularCoefficient = GeometricUtils.angularCoefficient(point1, point2)
101 
102     // if the straight line is vertical, the distance is the difference from myPoint X and a point on straight line X
103     if (angularCoefficient.isPosInfinity) Math.abs(myPoint.x - point1.x)
104     else Math.abs(myPoint.y - (angularCoefficient * myPoint.x + ordinateAtOrigin(point1, point2))) /
105       Math.sqrt(angularCoefficient.squared + 1)
106   }

108   /**
109     * A method that says if provided point is inside the circumference with provided center and radius
110     *
111     * @param point  the point to verify inside circumference
112     * @param center the center of the circumference
113     * @param radius the radius of the circumference
114     * @return true if the point is inside or on the circumference, false otherwise
115     */
116   def isWithinCircumference(point: Point,
117                             center: Point,
118                             radius: Double): Boolean = {
119     require(radius >= 0, "Circumference radius must be positive")
120 
121     (point.x - center.x).squared + (point.y - center.y).squared <= radius.squared
122   }
