134   /**
135     * @return the behaviour that manages rooms commands from GUI
136     */
137   private def roomsGUIBehaviour: Receive = {
138     case GUICreate(roomName, playersNumber) =>
139       log.info(s"Creating the room $roomName")
140       apiClientActor ! ServiceCreate(roomName, playersNumber, jwtToken)
141     case GUIEnterPrivate(roomID) =>
142       log.info(s"Entering the private room $roomID")
143       openOneTimeServerAndGetAddress()
144         .map(url => apiClientActor ! ServiceEnterPrivate(roomID, Address(playerAddress), url, jwtToken))
145     case GUIEnterPublic(playersNumber) =>
146       log.info(s"Entering the public room with $playersNumber players")
147       openOneTimeServerAndGetAddress()
148         .map(url => apiClientActor ! ServiceEnterPublic(playersNumber, Address(playerAddress), url, jwtToken))
149     case GUIExitPrivate(roomID) =>
150       log.info(s"Exiting room $roomID")
151       apiClientActor ! ServiceExitPrivate(roomID, jwtToken)
152     case GUIExitPublic(playersNumber) =>
153       log.info(s"Exiting public room with $playersNumber players")
154       apiClientActor ! ServiceExitPublic(playersNumber, jwtToken)
155     case GUILogOut =>
156       log.info("Logging-out")
157       onLogOut()
158   }

160   /**
161     * @return the behaviour that manages room API responses
162     */
163   // scalastyle:off cyclomatic.complexity
164   private def roomsApiReceiverBehaviour: Receive = {

166     case CreateFailure(errorMessage) => roomViewActor ! Error(CREATE_ERROR_TITLE, errorMessage.getOrElse(UNKNOWN_ERROR))

171     case EnterPublicFailure(errorMessage) => onRoomEnteringFailure(errorMessage)

173     case ExitPrivateSuccess => onRoomExitingSuccess()
174     case ExitPublicSuccess => onRoomExitingSuccess()
175     case ExitPrivateFailure(errorMessage) => onRoomExitingFailure(errorMessage)
176     case ExitPublicFailure(errorMessage) => onRoomExitingFailure(errorMessage)
177   }

224   /**

227     * @param errorMessage optionally an error message

229   private def onRoomExitingFailure(errorMessage: Option[String]): Unit = {
230     log.warn("Failed exiting the room, maybe participant number reached while trying to exit")
231     log.error(s"${errorMessage.getOrElse(UNKNOWN_ERROR)}")
232   }
233 
234   /**
235     * @return the Future containing the address of one-time server that will receive the participants
236     */
237   private def openOneTimeServerAndGetAddress(): Future[Address] = {
238     val onListReceived: List[Participant] => Unit = participants => {
239       log.info(s"Participants list received!")
240       onSuccessFindingOpponents(participants)
241     }
242 
243     log.debug(s"Starting the local one-time receiver server...")
244     listenForParticipantListFuture(onListReceived)
245       .andThen {
246         case Success(address) =>
247           log.debug(s"Server started and listening at the address: $address")
248         case Failure(error) =>
249           log.error(s"Problem starting the server: ${error.getMessage}")
250           roomViewActor ! Error(RECEIVING_PARTICIPANT_LIST_ERROR_TITLE, error.getMessage)
251       }
252   }
253 
254   /**
255     * Action to execute when found opponents
256     *
257     * @param participants the participants to game
258     */
259   private def onSuccessFindingOpponents(participants: List[Participant]): Unit = {
260     roomViewActor ! FoundOpponents
261     log.info(s"Setting the behaviour 'in-game'")
262     context.become(Actor.emptyBehavior)
263     roomViewActor ! Hide
264     playerActor ! PrepareForGame(participants,
265       CellWorldGenerationStrategy(GameViewActor.VIEW_SIZE, GameViewActor.VIEW_SIZE, GameConstants.PASSIVE_CELLS_NUMBER))
266   }
