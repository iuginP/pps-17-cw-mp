17 /**
18   * This actor manages execution of distributed game.
19   *
20   * It creates a cluster with other participants and maintains a distributed game state
21   *
22   * @author Eugenio Pierfederici
23   */
24 case class PlayerActor() extends Actor with Stash with Logging {
25 
26   // View actor
27   private var gameViewActor: ActorRef = _
28   private var playerName: String = _
29 
30   // Cluster info
31   private var roomSize: Int = _
32 
33   // distributed replica system
34   private val replicator: ActorRef = DistributedData(context.system).replicator
35   private val cluster: Cluster = Cluster(context.system)
36 
37   // Distributed world
38   private val distributedState: DistributedState[CellWorld] =
39     CellWorldDistributedState(onWorldUpdatedAction)(replicator, cluster)
40 
41   override def preStart(): Unit = {
42     log.info(s"Initializing the game-view actor...")
43     gameViewActor = context.system.actorOf(Props[GameViewActor], GameViewActor.getClass.getName)
44     gameViewActor ! Initialize
45 
46     log.info(s"Subscribing to cluster changes...")
47     cluster.subscribe(self, initialStateMode = InitialStateAsEvents,
48       classOf[MemberEvent], classOf[UnreachableMember])
49 
50     log.info("Subscribing to distributed state changes")
51     distributedState.subscribe(self)
52   }
53 
54   override def postStop(): Unit = {
55     cluster.unsubscribe(self)
56     distributedState.unsubscribe(self)
57   }
58 
59   override def receive: Receive = clusterBehaviour orElse beforeInGameBehaviour
60 
61   /**
62     * @return the behaviour to have before entering the game
63     */
64   private def beforeInGameBehaviour: Receive = {
65     case RetrieveAddress => sender() ! RetrieveAddressResponse(getAddress)
66 
67     case PrepareForGame(participants, worldGenerationStrategy) =>
68       roomSize = participants.size
69       cluster.join(AddressFromURIString(participants.head.address)) // all join first player cluster
70       if (getAddress == participants.head.address) { // first player injects start world
71         distributedState.initialize(worldGenerationStrategy(participants))
72       }
73       playerName = participants.find(participant => participant.address == getAddress).get.username
74 
75     case _ => stash() // stash distributed change until all players enter the room
76   }
77 
78   /**
79     * @return the behaviour of an actor in a cluster
80     */
81   private def clusterBehaviour: Receive = {
82     case MemberUp(member) =>
83       log.info("Member is Up: {}", member.address)
84       log.debug("Cluster size: " + cluster.state.members.size)
85       if (cluster.state.members.size == roomSize) startGame()
86     case UnreachableMember(member) =>
87       log.info("Member detected as unreachable: {}", member)
88     case MemberRemoved(member, previousStatus) =>
89       log.info("Member is Removed: {} after {}", member.address, previousStatus)
90     case _: MemberEvent => // ignore
91   }
92 
93   /**
94     * Starts the game
95     */
96   private def startGame(): Unit = {
97     context.become(inGameBehaviour)
98     gameViewActor ! ShowGUIWithName(playerName)
99     unstashAll() // un-stash distributed change messages
100   }
101 
102   /**
103     * @return the behaviour of the actor when it's in game
104     */
105   private def inGameBehaviour: Receive =
106     distributedState.distributedStateBehaviour orElse {
107       // TODO: remove this part of receive because we cannot come back to rooms view without stopping acotrSystem
108       case GameEnded => backToLobbyAction()
109     }
110 
111   /**
112     * The action to do when the game is ended
113     */
114   private def backToLobbyAction(): Unit = context.become(beforeInGameBehaviour orElse clusterBehaviour)
115 
116   /**
117     * Action that will be executed every time the world will be updated
118     *
119     * @param world the updated world
120     */
121   private def onWorldUpdatedAction(world: CellWorld): Unit = gameViewActor ! NewWorld(world)
122 
123   /**
124     * @return the address of this player
125     */
126   private def getAddress: String = cluster.selfAddress + self.path.toString.substring(self.path.address.toString.length)
127 }
