18 /**
19   * This actor manages execution of distributed game.
20   *
21   * It creates a cluster with other participants and maintains a distributed game state
22   *
23   * @author Eugenio Pierfederici
24   */
25 case class PlayerActor() extends Actor with Stash with Logging {
26 
27   // View actor
28   private var gameViewActor: ActorRef = _
29   private var playerName: String = _
30 
31   // Cluster info
32   private var roomSize: Int = _
33 
34   // distributed replica system
35   private val replicator: ActorRef = DistributedData(context.system).replicator
36   private val cluster: Cluster = Cluster(context.system)
37 
38   // Distributed world
39   private val distributedState: AkkaDistributedState[CellWorld] =
40     MergingStateCellWorld(onWorldUpdatedAction)(replicator, cluster)
41 
42   override def preStart(): Unit = {
43     log.info(s"Initializing the game-view actor...")
44     gameViewActor = context.system.actorOf(Props[GameViewActor], GameViewActor.getClass.getName)
45     gameViewActor ! Initialize
46 
47     log.info(s"Subscribing to cluster changes...")
48     cluster.subscribe(self, initialStateMode = InitialStateAsEvents,
49       classOf[MemberEvent], classOf[UnreachableMember])
50 
51     log.info("Subscribing to distributed state changes")
52     distributedState.subscribe(self)
53   }
54 
55   override def postStop(): Unit = {
56     cluster.unsubscribe(self)
57     distributedState.unsubscribe(self)
58   }
59 
60   override def receive: Receive = clusterBehaviour orElse beforeInGameBehaviour
61 
62   /**
63     * @return the behaviour to have before entering the game
64     */
65   private def beforeInGameBehaviour: Receive = {
66     case RetrieveAddress => sender() ! RetrieveAddressResponse(getAddress)
67 
68     case PrepareForGame(participants, worldGenerationStrategy) =>
69       roomSize = participants.size
70       cluster.join(AddressFromURIString(participants.head.address)) // all join first player cluster
71       if (getAddress == participants.head.address) { // first player injects start world
72         distributedState.initialize(worldGenerationStrategy(participants))
73       }
74       playerName = participants.find(participant => participant.address == getAddress).get.username
75 
76     case _ => stash() // stash distributed change until all players enter the room
77   }
78 
79   /**
80     * @return the behaviour of an actor in a cluster
81     */
82   private def clusterBehaviour: Receive = {
83     case MemberUp(member) =>
84       log.info("Member is Up: {}", member.address)
85       log.debug("Cluster size: " + cluster.state.members.size)
86       if (cluster.state.members.size == roomSize) startGame()
87     case UnreachableMember(member) =>
88       log.info("Member detected as unreachable: {}", member)
89     case MemberRemoved(member, previousStatus) =>
90       log.info("Member is Removed: {} after {}", member.address, previousStatus)
91     case _: MemberEvent => // ignore
92   }
93 
94   /**
95     * Starts the game
96     */
97   private def startGame(): Unit = {
98     context.become(inGameBehaviour)
99     gameViewActor ! ShowGUIWithName(playerName)
100     unstashAll() // un-stash distributed change messages
101   }
102 
103   /**
104     * @return the behaviour of the actor when it's in game
105     */
106   private def inGameBehaviour: Receive =
107     distributedState.distributedStateBehaviour orElse {
108       // TODO: remove this part of receive because we cannot come back to rooms view without stopping acotrSystem
109       case GameEnded => backToLobbyAction()
110     }
111 
112   /**
113     * The action to do when the game is ended
114     */
115   private def backToLobbyAction(): Unit = context.become(beforeInGameBehaviour orElse clusterBehaviour)
116 
117   /**
118     * Action that will be executed every time the world will be updated
119     *
120     * @param world the updated world
121     */
122   private def onWorldUpdatedAction(world: CellWorld): Unit = gameViewActor ! NewWorld(world)
123 
124   /**
125     * @return the address of this player
126     */
127   private def getAddress: String = cluster.selfAddress + self.path.toString.substring(self.path.address.toString.length)
128 }
