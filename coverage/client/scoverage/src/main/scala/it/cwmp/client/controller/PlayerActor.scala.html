31   // distributed replica system

33   private val cluster: Cluster = Cluster(context.system)

35   // Distributed world

39   override def preStart(): Unit = {
40     log.info(s"Initializing the game-view actor...")
41     gameViewActor = system.actorOf(Props(classOf[GameViewActor], self), "game-view")
42     log.info(s"Subscribing to cluster changes...")
43     cluster.subscribe(self, initialStateMode = InitialStateAsEvents,
44       classOf[MemberEvent], classOf[UnreachableMember])
45     distributedState.subscribe(self)
46   }
47 
48   override def postStop(): Unit = {
49     cluster.unsubscribe(self)
50     distributedState.unsubscribe(self)
51   }

53   override def receive: Receive = clusterBehaviour orElse lobbyBehaviour
54 
55   /**
56     * @return the behaviour of an actor in a cluster
57     */
58   private def clusterBehaviour: Receive = {
59     case MemberUp(member) =>
60       log.info("Member is Up: {}", member.address)
61       log.debug("Cluster size: " + cluster.state.members.size)
62       if (cluster.state.members.size == roomSize) enterGameAction()
63     case UnreachableMember(member) =>
64       log.info("Member detected as unreachable: {}", member)
65     case MemberRemoved(member, previousStatus) =>
66       log.info("Member is Removed: {} after {}", member.address, previousStatus)
67     case _: MemberEvent => // ignore
68   }
69 
70   // TODO: add doc
71   private def lobbyBehaviour: Receive = {
72     case RetrieveAddress =>
73       sender() ! RetrieveAddressResponse(getAddress)
74     case StartGame(participants) =>
75       join(participants)
76       roomSize = participants.size
77   }
78 
79   /**
80     * @return the behaviour of the actor when it's in game
81     */
82   private def inGameBehaviour: Receive =
83     distributedState.distributedStateBehaviour orElse {
84       case EndGame => backToLobbyAction()
85     }
86 
87   private def enterGameAction(): Unit = {
88     context.become(clusterBehaviour orElse inGameBehaviour)
89     gameViewActor ! ShowGUI
90     gameViewActor ! NewWorld(GameMain.debugWorld)
91   }
92 
93   private def backToLobbyAction(): Unit = context.become(clusterBehaviour orElse lobbyBehaviour)
94 
95   /**
96     * Action that will be executed every time the world will be updated
97     *
98     * @param world the updated world
99     */
100   private def onWorldUpdatedAction(world: CellWorld): Unit = gameViewActor ! NewWorld(world)
101 
102 
103   private def join(participants: List[Address]): Unit = {
104     cluster.join(AddressFromURIString(participants.head.address))
105   }
106 
107   private def getAddress: String = cluster.selfAddress + self.path.toString.substring(self.path.address.toString.length)
108 }
