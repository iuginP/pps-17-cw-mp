19   /**
20     * Evolves the world using default evolution strategies
21     *
22     * @param oldToEvolve the world to evolve
23     * @param elapsedTime the amount of time that should be reflected in changes in "toEvolve" world
24     * @return the evolved world
25     */
26   override def apply(oldToEvolve: CellWorld, elapsedTime: Duration): CellWorld = {
27     requireNonNull(oldToEvolve, "World to evolve must not be null")
28     requireNonNull(elapsedTime, "Elapsed time must not be null")
29 
30     implicit val elapsedTimeImplicit: Duration = elapsedTime
31     implicit val oldWorldInstant: Instant = oldToEvolve.instant
32 
33     val evolvedCellsAndCanAttack =
34       oldToEvolve.characters.map(naturallyEvolveCell(_, elapsedTime)) // natural energy evolution
35         .map(attackConsequencesOnAttackers(_, oldToEvolve.attacks)) // attackers reduction of energy, and not enough energy calculation
36 
37     var tempWorld = checkCellCanAttackAndRemoveNotPossibleAttacks(oldToEvolve, evolvedCellsAndCanAttack)
38 
39     // under attack reduction/increment of energy, and conquer management
40     val evolvedCells = tempWorld.characters.map(attackConsequencesOnAttacked(_, tempWorld.attacks))
41 
42     tempWorld = cancelPendingAttacksOfConqueredCells(tempWorld, evolvedCells)
43     tempWorld = updateAttacksToConqueredCells(tempWorld, tempWorld.characters)
44 
45     CellWorld(tempWorld.instant.plus(elapsedTime), tempWorld.characters, tempWorld.attacks)
46   }

48   /**
49     * Naturally evolves cell according to elapsed time
50     *
51     * @param cell        the cell to evolve
52     * @param elapsedTime the time elapsed
53     * @return the evolved cell
54     */
55   private def naturallyEvolveCell(cell: Cell, elapsedTime: Duration): Cell = {
56     if (Cell.isPassiveCell(cell)) {
57       cell.evolve(elapsedTime, Cell.Passive.defaultEvolutionStrategy)
58     } else {
59       cell.evolve(elapsedTime)
60     }

63   /**
64     * Calculation of attack consequences on attackers
65     *
66     * Each attacker loses part of its energy while a tentacle is travelling towards another cell
67     *
68     * @return the pair composed by the cell and a boolean indicating if that call can still attack or not (energy finished)
69     */
70   private def attackConsequencesOnAttackers(cell: Cell, tentacles: Seq[Tentacle])
71                                            (implicit oldWorldInstant: Instant, elapsedTime: Duration): (Cell, Boolean) = {
72 
73     var cellAndCanAttack = (cell, true)
74 
75     // all tentacles leaving from cell, checking if can still attack
76     for (tentacle <- tentacles if Cell.ownerAndPositionMatch(tentacle.from, cell) && cellAndCanAttack._2) {
77       val addedTentacleLength = tentacle.length(oldWorldInstant.plus(elapsedTime)) - tentacle.length(oldWorldInstant)
78       val attackerEnergyReduction = CellWorld.lengthToEnergyReductionStrategy(addedTentacleLength)
79 
80       if (cellAndCanAttack._1.energy <= attackerEnergyReduction) {
81         cellAndCanAttack = (cellAndCanAttack._1, false) // if no more energy, the cell should not attack anymore
82       } else {
83         cellAndCanAttack = (cellAndCanAttack._1 -- attackerEnergyReduction, cellAndCanAttack._2)
84       }
85     }
86     cellAndCanAttack
87   }

89   /**
90     * A method that takes a world in input and a specification of which cells can attack an which not,
91     * evolving accordingly the world
92     *
93     * @param world             the world to evolve
94     * @param cellsAndCanAttack a sequence of pairs where every cell is said to be able to attack or not
95     * @return the evolved world where cells that can attack are still attacking,
96     *         cell that cannot attack (because of not enough energy), are refunded of the spent one
97     *         and its attacks deleted from world
98     */
99   private def checkCellCanAttackAndRemoveNotPossibleAttacks(world: CellWorld,
100                                                             cellsAndCanAttack: Seq[(Cell, Boolean)]): CellWorld = {
101     var tempWorld = CellWorld(world.instant, cellsAndCanAttack.map(_._1), world.attacks)
102     val cellsThatCannotAffordAttacking = cellsAndCanAttack.filterNot(_._2).map(_._1)
103     for (poorCell <- cellsThatCannotAffordAttacking
104          if world.attacks.map(_.from).exists(Cell.ownerAndPositionMatch(_, poorCell))) {
105       // if cell cannot attack, refund it its energy and remove its last attack from world
106       tempWorld = tempWorld -- world.attacks
107         .filter(t => Cell.ownerAndPositionMatch(t.from, poorCell))
108         .max(Tentacle.orderByLaunchInstant)
109       // log.info(s"poorCell: $poorCell, removing ${world.attacks.filter(t => Cell.ownerAndPositionMatch(t.from, poorCell))
110       // .max(Tentacle.orderByLaunchInstant)}, after: $tempWorld")
111     }
112     tempWorld
113   }

115   /**
116     * Calculation of attack consequences on attacked cells
117     *
118     * If attacked cell is property of attacker the attack heals the cell
119     * If attacked cell is property of other player the attack damages the cell
120     *
121     * @return the updated cell
122     */
123   private def attackConsequencesOnAttacked(cell: Cell, allTentacles: Seq[Tentacle])
124                                           (implicit oldWorldInstant: Instant, elapsedTime: Duration): Cell = {
125     var toReturnCell = cell
126     val tentaclesToAttackedCell = allTentacles.filter(tentacle => Cell.ownerAndPositionMatch(tentacle.to, toReturnCell))
127     val actualInstant = oldWorldInstant.plus(elapsedTime)
128 
129     for (tentacle <- tentaclesToAttackedCell) {
130       val actualAttackDuration = tentacle.hasReachedDestinationFor(actualInstant)
131       val oldAttackDuration = tentacle.hasReachedDestinationFor(oldWorldInstant)
132       val addedAttackDuration = actualAttackDuration.minus(oldAttackDuration)
133 
134       val energyDelta = CellWorld.durationToEnergyConversionStrategy(addedAttackDuration)
135 
136       if (toReturnCell.owner == tentacle.from.owner) {
137         // owner of the cell is attacking his own cells -> heals the cell
138         toReturnCell = toReturnCell ++ energyDelta
139 
140       } else if (!Cell.isPassiveCell(toReturnCell)) {
141         // attacking others cell -> damages the cell
142 
143         if (toReturnCell.energy - energyDelta <= 0) {
144           // if cell reaches 0 energy is conquered
145 
146           toReturnCell = cellAfterConquer(toReturnCell, tentaclesToAttackedCell, actualInstant,
147             GameConstants.CELL_ENERGY_WHEN_BORN + (energyDelta - toReturnCell.energy))
148         } else {
149           toReturnCell = toReturnCell -- energyDelta
150         }
151       } else {
152         // attacking passive cell
153         if (actualAttackDuration.toMillis >= GameConstants.MILLIS_TO_PASSIVE_CELL_CONQUER) {
154           toReturnCell = cellAfterConquer(toReturnCell, tentaclesToAttackedCell, actualInstant, toReturnCell.energy)
155         }
156       }
157     }
158     toReturnCell
159   }

162   /**
163     * Defines how to behave after a cell conquer
164     *
165     * @param cell                    the cell that has been conquered
166     * @param tentaclesToAttackedCell the active tentacles attacking provided cell
167     * @param conquerInstant          the instant when conquer occurred
168     * @param newEnergy               the energy the cell should have
169     * @return the new cell after conquer
170     */
171   private def cellAfterConquer(cell: Cell,
172                                tentaclesToAttackedCell: Seq[Tentacle],
173                                conquerInstant: Instant,
174                                newEnergy: Double): Cell = {
175     val firstAttackerOfCell = tentaclesToAttackedCell.max(Tentacle.orderByAttackDuration(conquerInstant)).from.owner
176     Cell(firstAttackerOfCell, cell.position, newEnergy)
177   }

179   /**
180     * A method that takes a world where to find if some cells that have been conquered had attacks ongoing
181     * In that case those are cancelled
182     *
183     * @param world the world to evolve
184     * @param cells the cells to use
185     * @return a world where all attacks come from actual cells
186     */
187   private def cancelPendingAttacksOfConqueredCells(world: CellWorld,
188                                                    cells: Seq[Cell]): CellWorld = {
189     var tempWorld = CellWorld(world.instant, cells, world.attacks)
190     for (tentacle <- tempWorld.attacks if !cells.exists(Cell.ownerAndPositionMatch(_, tentacle.from))) {
191       tempWorld = CellWorld(world.instant, cells, tempWorld.attacks.filterNot(_ == tentacle))
192     }
193     tempWorld
194   }

196   /**
197     * A method that updates all attacks that were directed to a cell that has been conquered
198     *
199     * @param world the world
200     * @param cells the evolved cells
201     * @return a world where all attacks directed to no more "existing" cells (intended as pair owner-position)
202     *         will be redirected to cells in same position, updating tentacles
203     */
204   private def updateAttacksToConqueredCells(world: CellWorld,
205                                             cells: Seq[Cell]): CellWorld = {
206 
207     var tempWorld = CellWorld(world.instant, cells, world.attacks)
208     for (tentacle <- tempWorld.attacks if !cells.exists(Cell.ownerAndPositionMatch(_, tentacle.to))) {
209       val destinationUpdate = cells.filter(_.position == tentacle.to.position).head
210       val updatedTentacle = Tentacle(tentacle.from, destinationUpdate, tentacle.launchInstant)
211       tempWorld = CellWorld(world.instant, cells, tempWorld.attacks.filterNot(_ == tentacle) :+ updatedTentacle)
212     }
213     tempWorld
214   }
