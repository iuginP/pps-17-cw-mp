18   /**
19     * Evolves the world using default evolution strategies
20     *
21     * @param oldToEvolve the world to evolve
22     * @param elapsedTime the amount of time that should be reflected in changes in "toEvolve" world
23     * @return the evolved world
24     */
25   override def apply(oldToEvolve: CellWorld, elapsedTime: Duration): CellWorld = {
26     requireNonNull(oldToEvolve, "World to evolve must not be null")
27     requireNonNull(elapsedTime, "Elapsed time must not be null")
28 
29     implicit val elapsedTimeImplicit: Duration = elapsedTime
30     implicit val oldWorldInstant: Instant = oldToEvolve.instant
31 
32     val evolvedCellsAndCanAttack =
33       oldToEvolve.characters.map(_.evolve(elapsedTime)) // natural energy evolution
34         .map(attackConsequencesOnAttackers(_, oldToEvolve.attacks)) // attackers reduction of energy, and not enough energy calculation
35 
36     var tempWorld = checkCellCanAttackAndRemoveNotPossibleAttacks(oldToEvolve, evolvedCellsAndCanAttack)
37 
38     // under attack reduction/increment of energy, and conquer management
39     val evolvedCells = tempWorld.characters.map(attackConsequencesOnAttacked(_, tempWorld.attacks))
40 
41     tempWorld = cancelPendingAttacksOfConqueredCells(tempWorld, evolvedCells)
42     tempWorld = updateAttacksToConqueredCells(tempWorld, tempWorld.characters)
43 
44     CellWorld(tempWorld.instant.plus(elapsedTime), tempWorld.characters, tempWorld.attacks)
45   }

47   /**
48     * Calculation of attack consequences on attacked cells
49     *
50     * If attacked cell is property of attacker the attack heals the cell
51     * If attacked cell is property of other player the attack damages the cell
52     *
53     * @return the updated cell
54     */
55   private def attackConsequencesOnAttacked(cell: Cell, allTentacles: Seq[Tentacle])
56                                           (implicit oldWorldInstant: Instant, elapsedTime: Duration): Cell = {
57 
58     var toReturnCell = cell
59     for (tentacle <- allTentacles if Cell.ownerAndPositionMatch(tentacle.to, toReturnCell); // all tentacles arriving to cell
60          actualAttackDuration = tentacle.hasReachedDestinationFor(oldWorldInstant.plus(elapsedTime));
61          oldAttackDuration = tentacle.hasReachedDestinationFor(oldWorldInstant);
62          addedAttackDuration = actualAttackDuration.minus(oldAttackDuration)) {
63 
64       val energyDelta = CellWorld.durationToEnergyConversionStrategy(addedAttackDuration)
65 
66       // owner of the cell is attacking his own cells -> heals the cell
67       if (cell.owner == tentacle.from.owner) toReturnCell = toReturnCell ++ energyDelta
68       // attacking others cell -> damages the cell
69       else {
70         if (toReturnCell.energy - energyDelta <= 0) { // if cell reaches 0 energy is conquered
71           toReturnCell = cellAfterConquer(cell, allTentacles, energyDelta - toReturnCell.energy)
72         } else toReturnCell = toReturnCell -- energyDelta
73       }
74     }
75     toReturnCell
76   }

78   /**
79     * Defines how to behave after a cell conquer
80     *
81     * @param cell         the cell that has been conquered
82     * @param allTentacles the active tentacles
83     * @return the new cell after conquer
84     */
85   private def cellAfterConquer(cell: Cell,
86                                allTentacles: Seq[Tentacle],
87                                maturedEnergy: Double): Cell = {
88     val firstAttackerOfCell = allTentacles.min(Tentacle.orderByLaunchInstant).from.owner
89     Cell(firstAttackerOfCell, cell.position, GameConstants.cellWhenBornEnergy + maturedEnergy)
90   }

92   /**
93     * Calculation of attack consequences on attackers
94     *
95     * Each attacker loses part of its energy while a tentacle is travelling towards another cell
96     *
97     * @return the pair composed by the cell and a boolean indicating if that call can still attack or not (energy finished)
98     */
99   private def attackConsequencesOnAttackers(cell: Cell, tentacles: Seq[Tentacle])
100                                            (implicit oldWorldInstant: Instant, elapsedTime: Duration): (Cell, Boolean) = {
101 
102     var cellAndCanAttack = (cell, true)
103 
104     // all tentacles leaving from cell, checking if can still attack
105     for (tentacle <- tentacles if Cell.ownerAndPositionMatch(tentacle.from, cell) && cellAndCanAttack._2;
106          addedTentacleLength = tentacle.length(oldWorldInstant.plus(elapsedTime)) - tentacle.length(oldWorldInstant);
107          attackerEnergyReduction = CellWorld.lengthToEnergyReductionStrategy(addedTentacleLength)) {
108 
109       if (cellAndCanAttack._1.energy <= attackerEnergyReduction)
110         cellAndCanAttack = (cellAndCanAttack._1, false) // if no more energy, the cell should not attack anymore
111       else
112         cellAndCanAttack = (cellAndCanAttack._1 -- attackerEnergyReduction, cellAndCanAttack._2)
113     }
114     cellAndCanAttack
115   }

118   /**
119     * A method that takes a world in input and a specification of which cells can attack an which not,
120     * evolving accordingly the world
121     *
122     * @param world             the world to evolve
123     * @param cellsAndCanAttack a sequence of pairs where every cell is said to be able to attack or not
124     * @return the evolved world where cells that can attack are still attacking,
125     *         cell that cannot attack (because of not enough energy), are refunded of the spent one
126     *         and its attacks deleted from world
127     */
128   private def checkCellCanAttackAndRemoveNotPossibleAttacks(world: CellWorld,
129                                                             cellsAndCanAttack: Seq[(Cell, Boolean)]): CellWorld = {
130     var tempWorld = CellWorld(world.instant, cellsAndCanAttack.map(_._1), world.attacks)
131     val cellsThatCannotAffordAttacking = cellsAndCanAttack.filterNot(_._2).map(_._1)
132     for (poorCell <- cellsThatCannotAffordAttacking
133          if world.attacks.map(_.from).exists(Cell.ownerAndPositionMatch(_, poorCell))) {
134       // if cell cannot attack, refund it its energy and remove its last attack from world
135       tempWorld = tempWorld -- world.attacks
136         .filter(t => Cell.ownerAndPositionMatch(t.from, poorCell))
137         .max(Tentacle.orderByLaunchInstant)
138       // println(s"poorCell: $poorCell, removing ${world.attacks.filter(t => Cell.ownerAndPositionMatch(t.from, poorCell)).max(Tentacle.orderByLaunchInstant)}, after: $tempWorld")
139     }
140     tempWorld
141   }

143   /**
144     * A method that takes a world where to find if some cells that have been conquered had attacks ongoing
145     * In that case those are cancelled
146     *
147     * @param world the world to evolve
148     * @param cells the cells to use
149     * @return a world where all attacks come from actual cells
150     */
151   private def cancelPendingAttacksOfConqueredCells(world: CellWorld,
152                                                    cells: Seq[Cell]): CellWorld = {
153     var tempWorld = CellWorld(world.instant, cells, world.attacks)
154     for (tentacle <- tempWorld.attacks if !cells.exists(Cell.ownerAndPositionMatch(_, tentacle.from))) {
155       tempWorld = CellWorld(world.instant, cells, tempWorld.attacks.filterNot(_ == tentacle))
156     }
157     tempWorld
158   }
159 
160   /**
161     * A method that updates all attacks that were directed to a cell that has been conquered
162     *
163     * @param world the world
164     * @param cells the evolved cells
165     * @return a world where all attacks directed to no more "existing" cells (intended as pair owner-position)
166     *         will be redirected to cells in same position, updating tentacles
167     */
168   private def updateAttacksToConqueredCells(world: CellWorld,
169                                             cells: Seq[Cell]): CellWorld = {
170 
171     var tempWorld = CellWorld(world.instant, cells, world.attacks)
172     for (tentacle <- tempWorld.attacks if !cells.exists(Cell.ownerAndPositionMatch(_, tentacle.to))) {
173       val destinationUpdate = cells.filter(_.position == tentacle.to.position).head
174       val updatedTentacle = Tentacle(tentacle.from, destinationUpdate, tentacle.launchInstant)
175       tempWorld = CellWorld(world.instant, cells, tempWorld.attacks.filterNot(_ == tentacle) :+ updatedTentacle)
176     }
177     tempWorld
178   }
