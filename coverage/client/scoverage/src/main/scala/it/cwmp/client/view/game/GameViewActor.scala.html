79   /**
80     * The behaviour of receiving world modifications from external source
81     */
82   private def newWorldBehaviour: Receive = {
83     case NewWorld(world) =>
84       tempWorld = world
85       restartSynchronizationSchedule(WAIT_TIME_BEFORE_AUTOMATIC_SYNCHRONIZATION)
86       if (updatingSchedule == null) {
87         updatingSchedule = context.system.scheduler
88           .schedule(TIME_BETWEEN_FRAMES, TIME_BETWEEN_FRAMES, self, UpdateGUI)(context.dispatcher)
89       }
90       gameEnded(world) match {
91         case Some(winnerName) if winnerName == playerName => self ! AlertMessages.Info(playerName + YOU_WON_TITLE, YOU_WON)
92         case Some(_) => self ! AlertMessages.Info(playerName + YOU_LOST_TITLE, YOU_LOST)
93         case None =>
94       }
95 
96     case UpdateGUI =>
97       Future {
98         tempWorld = GameEngine(tempWorld, java.time.Duration.ofMillis(TIME_BETWEEN_FRAMES.toMillis))
99         tempWorld
100       } andThen {
101         case Success(cellWorld) => runOnUIThread { () => gameFX.updateWorld(cellWorld) }
102         case Failure(ex) =>
103           updatingSchedule.cancel()
104           log.error("Error calculating next CellWorld", ex)
105       }
106   }

108   /**
109     * The behaviour of listening for user events on GUI
110     */
111   private def guiWorldModificationsBehaviour: Receive = {
112     case AddAttack(from, to) =>
113       log.info(s"AddAttack from:$from to:$to")
114       val worldCharacters = tempWorld.characters
115       val fromCell = findCellNearTo(from, worldCharacters)
116       val toCell = findCellNearTo(to, worldCharacters)
117       (fromCell, toCell) match {
118         case (Some(attacker), Some(attacked)) if canAddAttack(attacker, attacked, tempWorld.attacks) =>
119           log.debug(s"Adding attack from $attacker to $attacked ...")
120           parentActor ! UpdateState(tempWorld ++ Tentacle(attacker, attacked, tempWorld.instant))
121           restartSynchronizationSchedule(WAIT_TIME_BEFORE_AUTOMATIC_SYNCHRONIZATION)
122         case tmp@_ => log.debug(s"No cells detected or auto-attack or not $playerName cell $tmp")
123       }
124 
125     case RemoveAttack(pointOnAttackView) =>
126       log.info(s"RemoveAttack pointOnView:$pointOnAttackView")
127       val attacks = findTentaclesNearTo(pointOnAttackView, tempWorld.attacks)
128       attacks.find(_.from.owner.username == playerName) match {
129         case Some(tentacle) =>
130           log.debug(s"Removing this attack: $tentacle ...")
131           parentActor ! UpdateState(tempWorld -- tentacle)
132           restartSynchronizationSchedule(WAIT_TIME_BEFORE_AUTOMATIC_SYNCHRONIZATION)
133         case tmp@_ => log.debug(s"No attack detected or not $playerName attack $tmp")
134       }
135 
136     case SynchronizeWorld =>
137       log.info("Time to synchronize other clients because of no user actions")
138       parentActor ! UpdateState(tempWorld)
139   }

141   /**
142     * Utility method to know if can add attack
143     *
144     * @param attacker       the attacker cell
145     * @param attacked       the attacked cell
146     * @param currentAttacks the current game attacks
147     * @return true if attack can be added, false otherwise
148     */
149   private def canAddAttack(attacker: Cell, attacked: Cell, currentAttacks: Seq[Tentacle]): Boolean = {
150     !Cell.ownerAndPositionMatch(attacker, attacked) && // no auto-attack
151       attacker.owner.username == playerName && // control only your cells
152       !currentAttacks.exists(tentacle => // no already present attacks
153         Cell.ownerAndPositionMatch(tentacle.from, attacker) &&
154           Cell.ownerAndPositionMatch(tentacle.to, attacked))
155   }
156 
157   /**
158     * Utility method to restart the world synchronization schedule
159     */
160   private def restartSynchronizationSchedule(waitTime: FiniteDuration): Unit = {
161     if (synchronizationSchedule != null) synchronizationSchedule.cancel()
162 
163     synchronizationSchedule = context.system.scheduler
164       .schedule(waitTime, waitTime, self, SynchronizeWorld)(context.dispatcher)
165   }

167   /**
168     * Checks if the game has ended
169     *
170     * @param cellWorld the cellWorld to check
171     * @return optionally the winner name
172     */
173   private def gameEnded(cellWorld: CellWorld): Option[String] =
174     if (cellWorld.characters.forall(_.owner == cellWorld.characters.head.owner)) {
175       cellWorld.characters.head.owner.username
176     } else None
177 }
