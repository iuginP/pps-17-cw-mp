14 /**
15   * The actor that deals with Game View
16   *
17   * @author contributor Enrico Siboni
18   */
19 class GameViewActor(parentActor: ActorRef) extends Actor with Logging {
20 
21   private val gameFX: GameFX = GameFX(self)
22   private val TIME_BETWEEN_FRAMES: FiniteDuration = 500.millis
23 
24   private var updatingSchedule: Cancellable = _
25   private var tempWorld: CellWorld = _
26 
27   override def receive: Receive = showGUIBehaviour
28 
29   /**
30     * The behaviour of opening the view
31     */
32   private def showGUIBehaviour: Receive = {
33     case ShowGUI =>
34       gameFX.start(VIEW_TITLE, VIEW_SIZE)
35       context.become(hideGUIBehaviour orElse
36         newWorldBehaviour orElse guiWorldModificationsBehaviour)
37   }
38 
39   /**
40     * The behaviour of closing the view
41     */
42   private def hideGUIBehaviour: Receive = {
43     case HideGUI =>
44       gameFX.close()
45       context.become(showGUIBehaviour)
46   }
47 
48   /**
49     * The behaviour of receiving world modifications from external source
50     */
51   private def newWorldBehaviour: Receive = {
52     case NewWorld(world) =>
53       if (updatingSchedule != null) updatingSchedule.cancel()
54       tempWorld = world
55       updatingSchedule = context.system.scheduler
56         .schedule(0.millis, TIME_BETWEEN_FRAMES, self, UpdateLocalWorld)(context.dispatcher)
57 
58     case UpdateLocalWorld =>
59       //  log.info(s"World to paint: Characters=${tempWorld.characters} Attacks=${tempWorld.attacks} Instant=${tempWorld.instant}")
60       gameFX.updateWorld(tempWorld)
61 
62       // is that to heavy computation here ???
63       tempWorld = GameEngine(tempWorld, java.time.Duration.ofMillis(TIME_BETWEEN_FRAMES.toMillis))
64   }
65 
66   /**
67     * The behaviour of listening for user events on GUI
68     */
69   private def guiWorldModificationsBehaviour: Receive = {
70     case AddAttack(from, to) =>
71       log.info(s"AddAttack from:$from to:$to")
72       val worldCharacters = tempWorld.characters
73       val fromCell = findCellNearTo(from, worldCharacters)
74       val toCell = findCellNearTo(to, worldCharacters)
75       (fromCell, toCell) match {
76         case (Some(attacker), Some(attacked)) if attacker != attacked =>
77           log.debug(s"Adding attack from $attacker to $attacked ...")
78           parentActor ! DistributedState.UpdateState(tempWorld ++ Tentacle(attacker, attacked, tempWorld.instant))
79         case tmp@_ => log.debug(s"No cells detected or auto-attack $tmp")
80       }
81 
82     case RemoveAttack(pointOnAttackView) =>
83       log.info(s"RemoveAttack pointOnView:$pointOnAttackView")
84       val attack = findTentacleNearTo(pointOnAttackView, tempWorld.attacks)
85       attack match {
86         case Some(tentacle) =>
87           log.debug(s"Removing this attack: $tentacle ...")
88           parentActor ! DistributedState.UpdateState(tempWorld -- tentacle)
89         case tmp@_ => log.debug(s"No attack detected $tmp")
90       }
91   }
92 }

148     * A method to find a cell near to a clicked point on view, according to actual cell sizing
149     *
150     * @param clickedPoint the clicked point on view
151     * @param cells        the collection of cells on screen
152     * @return optionally the cell near the clicked point
153     */
154   private def findCellNearTo(clickedPoint: Point, cells: Seq[Cell]): Option[Cell] =
155     cells.find(cell => GeometricUtils.isWithinCircumference(clickedPoint, cell.position, CellView.sizingStrategy(cell)))
156 
157   /**
158     * A method to find the tentacle near to a clicked point on view, according to actual tentacle sizing
159     *
160     * @param clickedPoint the cliked point on view
161     * @param tentacles    the collection of attacks on screen
162     * @return optionally the tentacle near the clicked point
163     */
164   private def findTentacleNearTo(clickedPoint: Point, tentacles: Seq[Tentacle]): Option[Tentacle] =
165     tentacles.find(tentacle => GeometricUtils.
166       pointDistanceFromStraightLine(clickedPoint, tentacle.from.position, tentacle.to.position) <= TentacleView.thicknessStrategy(tentacle))
