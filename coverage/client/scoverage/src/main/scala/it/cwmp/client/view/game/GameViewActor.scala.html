24 case class GameViewActor() extends Actor with Logging {
25 
26   private val gameFX: GameFX = GameFX(self)
27   private val TIME_BETWEEN_FRAMES: FiniteDuration = 350.millis
28 
29   private var parentActor: ActorRef = _
30   private var updatingSchedule: Cancellable = _
31   private var tempWorld: CellWorld = _
32   private var playerName: String = _
33 
34   override def receive: Receive = showGUIBehaviour orElse {
35     case Initialize => parentActor = sender()
36   }
37 
38   /**
39     * The behaviour of opening the view
40     */
41   private def showGUIBehaviour: Receive = {
42     case ShowGUIWithName(name) =>
43       playerName = name
44       gameFX.start(s"$VIEW_TITLE_PREFIX$name", VIEW_SIZE)
45       context.become(hideGUIBehaviour orElse newWorldBehaviour orElse guiWorldModificationsBehaviour)
46   }
47 
48   /**
49     * The behaviour of closing the view
50     */
51   private def hideGUIBehaviour: Receive = { // TODO: remove, no-one ever sends this message here
52     case Hide =>
53       if (updatingSchedule != null) updatingSchedule.cancel()
54       gameFX.close()
55       context.become(showGUIBehaviour)
56   }
57 
58   /**
59     * The behaviour of receiving world modifications from external source
60     */
61   private def newWorldBehaviour: Receive = {
62     case NewWorld(world) =>
63       tempWorld = world
64       if (updatingSchedule == null) {
65         updatingSchedule = context.system.scheduler
66           .schedule(TIME_BETWEEN_FRAMES, TIME_BETWEEN_FRAMES, self, UpdateGUI)(context.dispatcher)
67       }
68 
69     case UpdateGUI =>
70       Future {
71         tempWorld = GameEngine(tempWorld, java.time.Duration.ofMillis(TIME_BETWEEN_FRAMES.toMillis))
72         tempWorld
73       } andThen {
74         case Success(cellWorld) => gameFX.updateWorld(cellWorld)
75         case Failure(ex) =>
76           updatingSchedule.cancel()
77           log.error("Error calculating next CellWorld", ex)
78       }
79   }
80 
81   /**
82     * The behaviour of listening for user events on GUI
83     */
84   private def guiWorldModificationsBehaviour: Receive = {
85     case AddAttack(from, to) =>
86       log.info(s"AddAttack from:$from to:$to")
87       val worldCharacters = tempWorld.characters
88       val fromCell = findCellNearTo(from, worldCharacters)
89       val toCell = findCellNearTo(to, worldCharacters)
90       (fromCell, toCell) match {
91         case (Some(attacker), Some(attacked)) if canAddAttack(attacker, attacked, tempWorld.attacks) =>
92           log.debug(s"Adding attack from $attacker to $attacked ...")
93           parentActor ! UpdateState(tempWorld ++ Tentacle(attacker, attacked, tempWorld.instant))
94         case tmp@_ => log.debug(s"No cells detected or auto-attack or not $playerName cell $tmp")
95       }
96 
97     case RemoveAttack(pointOnAttackView) =>
98       log.info(s"RemoveAttack pointOnView:$pointOnAttackView")
99       val attack = findTentacleNearTo(pointOnAttackView, tempWorld.attacks)
100       attack match {
101         case Some(tentacle) if tentacle.from.owner.username == playerName =>
102           log.debug(s"Removing this attack: $tentacle ...")
103           parentActor ! UpdateState(tempWorld -- tentacle)
104         case tmp@_ => log.debug(s"No attack detected or not $playerName attack $tmp")
105       }
106   }
107 
108   /**
109     * Utility method to know if can add attack
110     *
111     * @param attacker       the attacker cell
112     * @param attacked       the attacked cell
113     * @param currentAttacks the current game attacks
114     * @return true if attack can be added, false otherwise
115     */
116   private def canAddAttack(attacker: Cell, attacked: Cell, currentAttacks: Seq[Tentacle]): Boolean = {
117     !Cell.ownerAndPositionMatch(attacker, attacked) && // no auto-attack
118       attacker.owner.username == playerName && // control only your cells
119       !currentAttacks.exists(tentacle => // no already present attacks
120         Cell.ownerAndPositionMatch(tentacle.from, attacker) &&
121           Cell.ownerAndPositionMatch(tentacle.to, attacked))
122   }
123 }

173     * A method to find a cell near to a clicked point on view, according to actual cell sizing
174     *
175     * @param clickedPoint the clicked point on view
176     * @param cells        the collection of cells on screen
177     * @return optionally the cell near the clicked point
178     */
179   private def findCellNearTo(clickedPoint: Point, cells: Seq[Cell]): Option[Cell] =
180     cells.find(cell => GeometricUtils.isWithinCircumference(clickedPoint, cell.position, CellView.sizingStrategy(cell)))
181 
182   /**
183     * A method to find the tentacle near to a clicked point on view, according to actual tentacle sizing
184     *
185     * @param clickedPoint the clicked point on view
186     * @param tentacles    the collection of attacks on screen
187     * @return optionally the tentacle near the clicked point
188     */
189   private def findTentacleNearTo(clickedPoint: Point, tentacles: Seq[Tentacle]): Option[Tentacle] =
190     tentacles.find(tentacle => GeometricUtils.
191       pointDistanceFromStraightLine(clickedPoint, tentacle.from.position, tentacle.to.position) <= TentacleView.thicknessStrategy(tentacle))
