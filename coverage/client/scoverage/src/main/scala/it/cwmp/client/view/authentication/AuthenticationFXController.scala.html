10 /**
11   * Trait that models the strategy to be applied to resolve authentication requests.
12   */
13 trait AuthenticationFXStrategy {
14   /**
15     * Function invoked for a system access request.
16     *
17     * @param username identification chosen by the player to access the system
18     * @param password password chosen during sign up
19     */
20   def onSignIn(username: String, password: String): Unit
21 
22   /**
23     * Function invoked for checking the correctness of the passwords.
24     *
25     * @param password password chosen
26     * @param confirmPassword confirmation password
27     * @return true, if the passwords respect the correctness policies
28     *         false, otherwise
29     */
30   def onCheckPassword(password: String, confirmPassword: String): Boolean
31 
32   /**
33     * Function invoked for a system registration request.
34     *
35     * @param username identification chosen by the player to register in the system
36     * @param password password chosen to authenticate in the system
37     */
38   def onSignUp(username: String, password: String): Unit
39 }
