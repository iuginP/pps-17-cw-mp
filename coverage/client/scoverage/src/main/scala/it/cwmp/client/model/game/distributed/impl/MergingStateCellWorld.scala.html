18 /**
19   * Distributed representation of CellWorld where modifications made concurrently are merged
20   *
21   * @param onWorldUpdate   the strategy to adopt on world changes
22   * @param replicatorActor the actor that will distribute the data
23   * @param cluster         the cluster where this distributed data are exchanged
24   * @author Enrico Siboni
25   */
26 case class MergingStateCellWorld(onWorldUpdate: CellWorld => Unit)(implicit replicatorActor: ActorRef, cluster: Cluster)
27   extends AkkaDistributedState[CellWorld](onWorldUpdate) {
28 
29   override type ReplicatedDataType = ORMultiMap[String, ReplicatedData]
30 
31   override protected val distributedKey: ORMultiMapKey[String, ReplicatedData] = ORMultiMapKey(DISTRIBUTED_KEY_NAME)
32 
33   override protected def activeBehaviour: Receive = {
34     case UpdateState(state: CellWorld) =>
35       log.debug("Updating distributed state")
36       updateDistributedStateTo(state)
37   }
38 
39   override protected def updateDistributedStateTo(state: CellWorld): Unit =
40     replicatorActor !
41       Update(distributedKey,
42         ORMultiMap.emptyWithValueDeltas[String, ReplicatedData],
43         consistencyPolicy
44       )(distributedModify(_, state))
45 
46   override protected def distributedModify(oldDistributedState: ORMultiMap[String, ReplicatedData],
47                                            newState: CellWorld): ORMultiMap[String, ReplicatedData] = {
48 
49     val distributedInstant = LWWRegister[Instant](newState.instant)
50     val distributedCharacters = newState.characters.foldLeft(ORSet.empty[Cell])(_ + _)
51     val distributedAttacks = newState.attacks.foldLeft(ORSet.empty[Tentacle])(_ + _)
52 
53     oldDistributedState.entries
54       .foldLeft(oldDistributedState)(_ - _._1) // removes all bindings
55       .addBinding(INSTANT_DISTRIBUTED_KEY, distributedInstant)
56       .addBinding(CELLS_DISTRIBUTED_KEY, distributedCharacters)
57       .addBinding(TENTACLE_DISTRIBUTED_KEY, distributedAttacks)
58   }
59 
60   override protected def parseFromDistributed(distributedData: ORMultiMap[String, ReplicatedData]): CellWorld = {
61     val instantOption = distributedData.get(INSTANT_DISTRIBUTED_KEY)
62     val charactersOption = distributedData.get(CELLS_DISTRIBUTED_KEY)
63     val attacksOption = distributedData.get(TENTACLE_DISTRIBUTED_KEY)
64 
65     (instantOption, charactersOption, attacksOption) match {
66       case (Some(replicatedInstant), Some(replicatedCharacters), Some(replicatedAttacks))
67         if replicatedInstant.nonEmpty && replicatedCharacters.nonEmpty && replicatedAttacks.nonEmpty =>
68 
69         val distributedInstant = replicatedInstant.head.asInstanceOf[LWWRegister[Instant]]
70         val distributedCharacters = replicatedCharacters.head.asInstanceOf[ORSet[Cell]]
71         val distributedAttacks = replicatedAttacks.head.asInstanceOf[ORSet[Tentacle]]
72 
73         CellWorld(distributedInstant.value, distributedCharacters.elements.toSeq, distributedAttacks.elements.toSeq)
74 
75       case cause@_ =>
76         val errorMessage = s"Cannot convert from distributed data: $cause"
77         log.error(errorMessage)
78         throw Utils.parseException("Converting from DistributedData to CellWorld", errorMessage)
79     }
80   }
81 }
