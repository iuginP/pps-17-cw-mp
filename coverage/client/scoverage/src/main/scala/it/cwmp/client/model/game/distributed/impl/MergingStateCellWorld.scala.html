19 /**
20   * Distributed representation of CellWorld where modifications made concurrently are merged
21   *
22   * @param onWorldUpdate   the strategy to adopt on world changes
23   * @param replicatorActor the actor that will distribute the data
24   * @param cluster         the cluster where this distributed data are exchanged
25   * @author Enrico Siboni
26   */
27 case class MergingStateCellWorld(onWorldUpdate: CellWorld => Unit)(implicit replicatorActor: ActorRef, cluster: Cluster)
28   extends AkkaDistributedState[CellWorld](onWorldUpdate) {
29 
30   override type ReplicatedDataType = ORMultiMap[String, ReplicatedData]
31 
32   override protected val distributedKey: ORMultiMapKey[String, ReplicatedData] = ORMultiMapKey(DISTRIBUTED_KEY_NAME)
33 
34   override protected def activeBehaviour: Receive = {
35     case UpdateState(state: CellWorld) =>
36       log.debug("Updating distributed state")
37       updateDistributedStateTo(state)
38   }
39 
40   override protected def updateDistributedStateTo(state: CellWorld): Unit =
41     replicatorActor !
42       Update(distributedKey,
43         ORMultiMap.emptyWithValueDeltas[String, ReplicatedData],
44         consistencyPolicy
45       )(distributedModify(_, state))
46 
47   override protected def distributedModify(oldDistributedState: ORMultiMap[String, ReplicatedData],
48                                            newState: CellWorld): ORMultiMap[String, ReplicatedData] = {
49     val distributedInstant = LWWRegister[Instant](
50       try {
51         val oldInstant = parseFromDistributed(oldDistributedState).instant
52         if (oldInstant.isBefore(newState.instant)) newState.instant else oldInstant
53       } catch {
54         case _: ParseException => newState.instant
55       }
56     )
57     val distributedCharacters = newState.characters.foldLeft(ORSet.empty[Cell])(_ + _)
58     val distributedAttacks = newState.attacks.foldLeft(ORSet.empty[Tentacle])(_ + _)
59 
60     oldDistributedState.entries
61       .foldLeft(oldDistributedState)(_ - _._1) // removes all bindings
62       .addBinding(INSTANT_DISTRIBUTED_KEY, distributedInstant)
63       .addBinding(CELLS_DISTRIBUTED_KEY, distributedCharacters)
64       .addBinding(TENTACLE_DISTRIBUTED_KEY, distributedAttacks)
65   }
66 
67   override protected def parseFromDistributed(distributedData: ORMultiMap[String, ReplicatedData]): CellWorld = {
68     val instantOption = distributedData.get(INSTANT_DISTRIBUTED_KEY)
69     val charactersOption = distributedData.get(CELLS_DISTRIBUTED_KEY)
70     val attacksOption = distributedData.get(TENTACLE_DISTRIBUTED_KEY)
71 
72     (instantOption, charactersOption, attacksOption) match {
73       case (Some(replicatedInstant), Some(replicatedCharacters), Some(replicatedAttacks))
74         if replicatedInstant.nonEmpty && replicatedCharacters.nonEmpty && replicatedAttacks.nonEmpty =>
75 
76         val distributedInstant = replicatedInstant.head.asInstanceOf[LWWRegister[Instant]]
77         val distributedCharacters = replicatedCharacters.head.asInstanceOf[ORSet[Cell]]
78         val distributedAttacks = replicatedAttacks.head.asInstanceOf[ORSet[Tentacle]]
79 
80         CellWorld(distributedInstant.value, distributedCharacters.elements.toSeq, distributedAttacks.elements.toSeq)
81 
82       case cause@_ =>
83         val errorMessage = s"Cannot convert from distributed data: $cause"
84         log.error(errorMessage)
85         throw Utils.parseException("Converting from DistributedData to CellWorld", errorMessage)
86     }
87   }
88 }
